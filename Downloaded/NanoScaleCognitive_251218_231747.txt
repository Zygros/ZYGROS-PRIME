        """Generate batch of nodes with genetic variation"""
        batch = []
        
        for i in range(count):
            # Create genetic variation
            mutated_dna = self._mutate_dna(seed.get("dna", ""))

                "id": self.node_counter,
                "dna": mutated_dna,
                "fitness": np.random.random(),
                "capabilities": seed.get("capabilities", {}),
                "memory": {"cycle": i},
                "parent_id": seed.get("id", 0)
            }
            
            batch.append(node)
            self.node_counter += 1
        
        self.stats["nodes_created"] += count
        return batch
    
    def _mutate_dna(self, dna: str) -> str:
        """Apply genetic mutations"""
        if len(dna) < 64:
            dna = dna.ljust(64, 'A')

        mutations = np.random.randint(0, 64, size=3)
        
        bases = ['A', 'T', 'C', 'G', 'Q']
        for pos in mutations:
            current = dna_list[pos]
            available = [b for b in bases if b != current]
            dna_list[pos] = np.random.choice(available)
        
        return ''.join(dna_list)

              f"Memory: {self.stats['memory_used_mb']:.1f} MB | "
              f"Ratio: {self.stats['compression_ratio']:.1f}:1", end="")
