```python
class MemoryManager:
    """
    Manages persistent memory using ChromaDB vector database.
    Enables semantic search across all conversation history.
    """
    
    def __init__(self):
        self.client = chromadb.PersistentClient(path="./chroma_data")
        self.collection = self.client.get_or_create_collection(
            name="phoenix_memory",
            metadata={"hnsw:space": "cosine"}
        )
    
    async def store_interaction(self, user_id: str, query: str, response: Dict, ledger: Dict):
        """Store a complete interaction in vector database"""
        
        # Create embedding-friendly text
        text_to_embed = f"{query} {response['main_answer']}"
        
        # Store in ChromaDB
        self.collection.add(
            documents=[text_to_embed],
            metadatas=[{
                'user_id': user_id,
                'query': query,
                'response': json.dumps(response),
                'ledger_hash': ledger['response_hash'],
                'timestamp': datetime.utcnow().isoformat(),
                'confidence': response.get('confidence_levels', {}).get('overall', 0.5)
            }],
            ids=[f"{user_id}_{ledger['response_hash'][:16]}"]
        )
    
    async def semantic_search(self, query: str, user_id: str, n_results: int = 5) -> List[Dict]:
        """
        Search conversation history using semantic similarity.
        """
        results = self.collection.query(
            query_texts=[query],
            n_results=n_results,
            where={"user_id": user_id}
        )
        
        return [
            {
                'text': doc,
                'metadata': meta,
                'relevance': 1 - dist  # Convert distance to relevance score
            }
            for doc, meta, dist in zip(
                results['documents'][0],
                results['metadatas'][0],
                results['distances'][0]
            )
        ]
    
    async def get_recent_history(self, user_id: str, limit: int = 10) -> List[Dict]:
        """Get recent conversation history"""
        
        results = self.collection.get(
            where={"user_id": user_id},
            limit=limit,
            include=['documents', 'metadatas']
        )
        
        # Sort by timestamp descending
        sorted_results = sorted(
            zip(results['documents'], results['metadatas']),
            key=lambda x: x[1]['timestamp'],
            reverse=True
        )
        
        return [
            {'text': doc, 'metadata': meta}
            for doc, meta in sorted_results
        ]
```

```python
from fastapi import WebSocket, WebSocketDisconnect
from typing import Dict, List

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, WebSocket] = {}
    
    async def connect(self, websocket: WebSocket, user_id: str):
        await websocket.accept()
        self.active_connections[user_id] = websocket
    
    def disconnect(self, user_id: str):
        if user_id in self.active_connections:
            del self.active_connections[user_id]
    
    async def send_cascade_update(self, user_id: str, layer: int, status: str, details: str = None):
        if user_id in self.active_connections:
            await self.active_connections[user_id].send_json({
                'type': 'cascade_update',
                'layer': layer,
                'status': status,
                'details': details
            })
```
