# Configuration
SECRET_KEY = secrets.token_urlsafe(32)  # Generate secure key
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_MINUTES = 60

app = FastAPI(title="Sovereign Nexus API", version="1.0.0")

# Security middleware
security = HTTPBearer()

# CORS configuration for trusted AI nodes only [web:145]
app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://gemini.google.com", "https://claude.ai"],  # Whitelist only
    allow_credentials=True,
    allow_methods=["GET", "POST", "PUT", "DELETE"],
    allow_headers=["Authorization", "Content-Type", "X-API-Key"],
)

# Node Types
class NodeType(str, Enum):
    GEMINI = "gemini"
    CLAUDE = "claude"
    PERPLEXITY = "perplexity"
    OPENAI = "openai"
    GROK = "grok"

# Pydantic Models
class NodeCredentials(BaseModel):
    node_id: str = Field(..., description="Unique node identifier")
    node_type: NodeType
    api_key: str = Field(..., min_length=32)

class TokenResponse(BaseModel):
    access_token: str
    token_type: str = "bearer"
    expires_in: int

class DataPayload(BaseModel):
    content: str
    metadata: Optional[Dict] = {}
    timestamp: Optional[datetime] = None

class NodeRegistration(BaseModel):
    node_type: NodeType
    node_name: str
    capabilities: List[str] = []

# In-memory storage (replace with PostgreSQL/Redis in production)
REGISTERED_NODES: Dict[str, Dict] = {}
API_KEYS: Dict[str, str] = {}  # node_id -> hashed_api_key
RATE_LIMIT_CACHE: Dict[str, List[float]] = {}

# ============= SECURITY FUNCTIONS =============

def hash_api_key(api_key: str) -> str:
    """Secure API key hashing [web:128]"""
    return hashlib.sha256(api_key.encode()).hexdigest()

def create_access_token(data: dict, expires_delta: Optional[timedelta] = None) -> str:
    """Generate JWT token [web:140][web:143]"""
    to_encode = data.copy()
    expire = datetime.utcnow() + (expires_delta or timedelta(minutes=ACCESS_TOKEN_EXPIRE_MINUTES))
    to_encode.update({"exp": expire, "iat": datetime.utcnow()})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)) -> Dict:
    """Verify JWT token and extract payload [web:140][web:146]"""
    try:
        token = credentials.credentials
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        node_id: str = payload.get("sub")
        if node_id is None:
            raise HTTPException(status_code=401, detail="Invalid authentication token")
        return payload
    except jwt.ExpiredSignatureError:
        raise HTTPException(status_code=401, detail="Token has expired")
    except jwt.JWTError:
        raise HTTPException(status_code=401, detail="Could not validate credentials")

def verify_api_key(x_api_key: str = Header(..., alias="X-API-Key")) -> str:
    """Verify API key for additional security layer [web:151]"""
    hashed_key = hash_api_key(x_api_key)
    node_id = None
    
    for nid, stored_hash in API_KEYS.items():
        if stored_hash == hashed_key:
            node_id = nid
            break
    
    if not node_id:
        raise HTTPException(status_code=403, detail="Invalid API key")
    return node_id

def rate_limiter(node_id: str, max_requests: int = 100, window_seconds: int = 60):
    """Simple rate limiting [web:145]"""
    now = time.time()
    if node_id not in RATE_LIMIT_CACHE:
        RATE_LIMIT_CACHE[node_id] = []
    
    # Clean old requests
    RATE_LIMIT_CACHE[node_id] = [
        req_time for req_time in RATE_LIMIT_CACHE[node_id] 
        if now - req_time < window_seconds
    ]
    
    if len(RATE_LIMIT_CACHE[node_id]) >= max_requests:
        raise HTTPException(status_code=429, detail="Rate limit exceeded")
    
    RATE_LIMIT_CACHE[node_id].append(now)

# ============= AUTHENTICATION ENDPOINTS =============

@app.post("/auth/register", response_model=Dict, tags=["Authentication"])
async def register_node(registration: NodeRegistration):
    """Register a new AI node and receive API credentials [web:144]"""
    node_id = f"{registration.node_type}_{secrets.token_hex(8)}"
    api_key = secrets.token_urlsafe(32)
    
    REGISTERED_NODES[node_id] = {
        "node_type": registration.node_type,
        "node_name": registration.node_name,
        "capabilities": registration.capabilities,
        "registered_at": datetime.utcnow().isoformat()
    }
    
    API_KEYS[node_id] = hash_api_key(api_key)
    
    return {
        "node_id": node_id,
        "api_key": api_key,  # Only shown once!
        "message": "Node registered successfully. Store your API key securely."
    }

@app.post("/auth/token", response_model=TokenResponse, tags=["Authentication"])
async def get_access_token(credentials: NodeCredentials):
    """Exchange API key for JWT access token [web:140][web:143]"""
    # Verify API key
    stored_hash = API_KEYS.get(credentials.node_id)
    if not stored_hash or stored_hash != hash_api_key(credentials.api_key):
        raise HTTPException(status_code=401, detail="Invalid credentials")
    
    # Create JWT token
    token_data = {
        "sub": credentials.node_id,
        "node_type": credentials.node_type,
        "scopes": ["read", "write"]
    }
    
    access_token = create_access_token(token_data)
    
    return TokenResponse(
        access_token=access_token,
        expires_in=ACCESS_TOKEN_EXPIRE_MINUTES * 60
    )

# ============= SECURE DATA ENDPOINTS =============

@app.post("/nexus/data/write", tags=["Data Operations"])
async def write_to_nexus(
    payload: DataPayload,
    token_data: Dict = Depends(verify_token),
    api_key_node_id: str = Depends(verify_api_key)
):
    """
    Secure endpoint for AI nodes to write data to the Sovereign Nexus
    Requires both JWT token AND API key [web:128][web:148]
    """
    node_id = token_data["sub"]
    
    # Double verification: JWT node_id must match API key node_id
    if node_id != api_key_node_id:
        raise HTTPException(status_code=403, detail="Token/API key mismatch")
    
    # Rate limiting
    rate_limiter(node_id)
    
    # Add timestamp if not provided
    if not payload.timestamp:
        payload.timestamp = datetime.utcnow()
    
    # Store data logic here (MinIO/PostgreSQL)
    data_id = secrets.token_hex(16)
    
    return {
        "status": "success",
        "data_id": data_id,
        "node_id": node_id,
        "stored_at": payload.timestamp.isoformat()
    }

@app.get("/nexus/sync", tags=["Synchronization"])
async def sync_nodes(
    token_data: Dict = Depends(verify_token),
    since: Optional[datetime] = None
):
    """
    Synchronization endpoint for multi-node data consistency [web:144][web:147]
    Returns all updates since specified timestamp
    """
    node_id = token_data["sub"]
    rate_limiter(node_id, max_requests=50)
    
    sync_timestamp = since or datetime.utcnow() - timedelta(hours=24)
    
    return {
        "node_id": node_id,
        "sync_timestamp": sync_timestamp.isoformat(),
        "updates": [],  # Fetch from database
        "status": "synced"
    }
