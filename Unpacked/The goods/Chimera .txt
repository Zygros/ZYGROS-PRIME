# --- TEXT-TO-SPEECH ENGINE ---
try:
    tts_engine = pyttsx3.init()
    voices = tts_engine.getProperty('voices')
    # Attempt to find a more natural, less robotic voice. This is OS-dependent.
    # On Windows, 'Zira' or 'David' are often better. On macOS, 'Daniel' or 'Samantha'.
    try:
        # Try to find a specific, high-quality voice
        preferred_voice = next((v for v in voices if 'zira' in v.name.lower() or 'david' in v.name.lower()), voices[0])
        tts_engine.setProperty('voice', preferred_voice.id)
    except:
        pass # Fallback to default
    tts_engine.setProperty('rate', 175)  # Slightly faster than default
    tts_engine.setProperty('volume', 1.0)
except Exception as e:
    print(f"[TTS ERROR] Could not initialize text-to-speech engine: {e}")
    tts_engine = None

# --- GUI CLASS ---
class ChimeraGUI:
    def __init__(self, root, send_callback):
        self.root = root
        self.send_callback = send_callback
        self.root.title(WINDOW_TITLE)
        self.root.configure(bg=BG_COLOR)

        # --- Face Animation ---
        self.face_canvas = tk.Canvas(root, width=150, height=150, bg=BG_COLOR, highlightthickness=0)
        self.face_canvas.pack(pady=20)
        self.faces = {
            "idle": self.create_face_image("o_o"),
            "thinking": self.create_face_image("o.o"),
            "speaking": self.create_face_image("^-^"),
            "listening": self.create_face_image("O_O"),
            "error": self.create_face_image("X_X")
        }
        self.current_face = self.face_canvas.create_image(75, 75, image=self.faces["idle"])

        # --- Chat History ---
        self.chat_history = scrolledtext.ScrolledText(root, wrap=tk.WORD, bg="#101018", fg=TEXT_COLOR, font=(FONT_FACE, 12), bd=0, relief="flat", padx=10, pady=10)
        self.chat_history.pack(padx=10, pady=10, fill=tk.BOTH, expand=True)
        self.chat_history.tag_config("architect", foreground="#87CEEB") # Light Blue
        self.chat_history.tag_config("chimera", foreground="#00ff41") # Neon Green
        self.chat_history.tag_config("system", foreground="#ff6b6b")  # Reddish
        self.chat_history.config(state=tk.DISABLED)

        # --- Input Box ---
        self.input_frame = tk.Frame(root, bg=BG_COLOR)
        self.input_frame.pack(fill=tk.X, padx=10, pady=10)
        
        self.input_field = tk.Entry(self.input_frame, bg="#2c2c3e", fg=TEXT_COLOR, font=(FONT_FACE, 14), insertbackground=TEXT_COLOR, bd=0)
        self.input_field.pack(side=tk.LEFT, fill=tk.X, expand=True, ipady=8)
        self.input_field.bind("<Return>", self.on_send)

        self.send_button = tk.Button(self.input_frame, text="Send", command=self.on_send, bg="#4a4a6a", fg=TEXT_COLOR, relief="flat", font=(FONT_FACE, 12, "bold"))
        self.send_button.pack(side=tk.RIGHT, padx=(5, 0))

    def create_face_image(self, text):
        image = Image.new("RGBA", (150, 150), (0, 0, 0, 0))
        from PIL import ImageDraw, ImageFont
        draw = ImageDraw.Draw(image)
        try:
            font = ImageFont.truetype("cour.ttf", 48)
        except IOError:
            font = ImageFont.load_default()
        
        bbox = draw.textbbox((0,0), text, font=font)
        text_width = bbox[2] - bbox[0]
        text_height = bbox[3] - bbox[1]
        
        draw.text(((150 - text_width) / 2, (150 - text_height) / 2), text, font=font, fill=ACCENT_COLOR)
        return ImageTk.PhotoImage(image)

    def set_face(self, face_name):
        if face_name in self.faces:
            self.face_canvas.itemconfig(self.current_face, image=self.faces[face_name])

    def on_send(self, event=None):
        user_input = self.input_field.get()
        if user_input.strip():
            self.send_callback(user_input)
            self.input_field.delete(0, tk.END)

    def add_message(self, sender, message, tag):
        self.chat_history.config(state=tk.NORMAL)
        self.chat_history.insert(tk.END, f"{sender}: {message}\n\n", tag)
        self.chat_history.config(state=tk.DISABLED)
        self.chat_history.see(tk.END)

# --- Main Application Logic ---
class ChimeraApp:
    def __init__(self, root):
        self.root = root
        self.gui = ChimeraGUI(root, self.process_input)
        self.message_queue = queue.Queue()
        self.conversation_history = [{"role": "system", "content": SYSTEM_PROMPT}]
        
        self.display_message("CHIMERA", "System Online. Awaiting decree, Architect.", "chimera")
        self.speak("System Online. Awaiting decree, Architect.")

    def process_input(self, user_input):
        self.display_message(f"{ARCHITECT_NAME}", user_input, "architect")
        self.gui.set_face("thinking")
        
        # Add user message to history
        self.conversation_history.append({"role": "user", "content": user_input})
        
        # Start AI processing in a separate thread to keep UI responsive
        threading.Thread(target=self.get_ai_response, args=(user_input,), daemon=True).start()
        self.root.after(100, self.check_queue)

    def get_ai_response(self, user_input):
        try:
            client = openai.OpenAI(api_key=os.getenv("OPENAI_API_KEY"))
            response = client.chat.completions.create(
                model="gpt-4o",  # The most capable model
                messages=self.conversation_history,
                temperature=0.5,
                max_tokens=4096
            )
            ai_response = response.choices[0].message.content
            self.conversation_history.append({"role": "assistant", "content": ai_response})
            self.message_queue.put(ai_response)
        except Exception as e:
            error_message = f"ERROR: Could not connect to the cognitive core. {str(e)}"
            self.message_queue.put(error_message)

    def check_queue(self):
        try:
            message = self.message_queue.get_nowait()
            if "ERROR:" in message:
                self.display_message("SYSTEM", message, "error")
                self.gui.set_face("error")
            else:
                self.display_message(AI_NAME, message, "chimera")
                self.speak(message)
                self.gui.set_face("idle")
        except queue.Empty:
            self.root.after(100, self.check_queue)

    def display_message(self, sender, message, tag):
        self.gui.add_message(sender, message, tag)

    def speak(self, text):
        if tts_engine:
            self.gui.set_face("speaking")
            try:
                tts_engine.say(text)
                tts_engine.runAndWait()
            except Exception as e:
                self.display_message("SYSTEM", f"TTS Error: {e}", "error")
            finally:
                self.gui.set_face("idle")

def main():
    root = tk.Tk()
    app = ChimeraApp(root)
    root.mainloop()

if __name__ == "__main__":
    main()
