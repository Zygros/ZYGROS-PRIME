-f "$PLUGIN" ]; then echo "❌ Expected plugin at $PLUGIN" echo " Save your MAIA code there and re-run: bash ~/install_maia_termux_full.sh" exit 1 fi # ───────────────────────────────────────────────────────────────────────────── # 4) Virtualenv (optional but clean) # ───────────────────────────────────────────────────────────────────────────── if ! -f "$PLUGIN" ]; then echo "❌ Missing $PLUGIN" >&2; exit 1; fi # If script(1) exists, run with logging when invoked directly. if command -v script >/dev/null 2>&1; then ts="$(date +%Y%m%d_%H%M%S)" exec script -q -c "$PY \"$PLUGIN\"" "$HOME/maia/logs/maia_${ts}.log" else exec "$PY" "$PLUGIN" fi ;; help|-h|--help) usage ;; status) shift "$PY" - <<'PY' import json, maia_universal_plugin as m print(json.dumps(m.maia_status(), indent=2)) PY ;; process) shift q="${*:-}" if [ -z "$q" ]; then echo "❌ Provide a query string." >&2; exit 1; fi "$PY" - <<PY import json, maia_universal_plugin as m m.activate_maia({"host":"Termux","device":"Android"}) print(json.dumps(m.maia_process("""$q"""), indent=2)) PY ;; operator) shift cmd="${*:-status}" "$PY" - <<PY import maia_universal_plugin as m m.activate_maia({"host":"Termux","device":"Android"}) m.maia_control("$cmd") print("OK") PY ;; doctor) "$PY" - <<'PY' import importlib, os, sys, json, pathlib root = os.path.expanduser("~/maia") res = {"plugin_exists": os.path.isfile(os.path.join(root,"maia_universal_plugin.py")), "python": sys.version.split()[0], "cwd": os.getcwd(), "py_search_root": root} try: m = importlib.import_module("maia_universal_plugin") res["import_ok"] = True res["has_activate_maia"] = hasattr(m,"activate_maia") res["has_maia_process"] = hasattr(m,"maia_process") res["has_maia_status"] = hasattr(m,"maia_status") except Exception as e: res["import_ok"] = False res["error"] = repr(e) print(json.dumps(res, indent=2)) PY ;; logtail) f="$(logfile_latest || true)" if [ -z "${f:-}" ]; then echo "No logs yet."; exit 0; fi tail -n 200 -f "$f" ;; *) usage; exit 1 ;; esac BASH chmod +x "$BIN_DIR/maia" # ───────────────────────────────────────────────────────────────────────────── # 6) PATH + Storage # ───────────────────────────────────────────────────────────────────────────── SHELL_RC="$HOME/.bashrc" [ -f "$HOME/.zshrc" ] && SHELL_RC="$HOME/.zshrc" if ! Ensure Termux:Boot is installed & opened once." fi # ───────────────────────────────────────────────────────────────────────────── # 8) Cron keepalive (optional fallback) # ───────────────────────────────────────────────────────────────────────────── INSTALL_CRON=0 if command -v crond >/dev/null 2>&1; then read -p "Enable cron keepalive (starts MAIA if killed)? [Y/n] " yn yn=${yn:-Y} if [[ "$yn" =~ ^[Yy]$ ]]; then INSTALL_CRON=1 fi fi if [ $INSTALL_CRON -eq 1 ]; then mkdir -p "$BOOT_DIR" cat > "$BOOT_DIR/crond_boot.sh" <<'BOOT' #!/data/data/com.termux/files/usr/bin/bash termux-wake-lock 2>/dev/null || true crond -b BOOT chmod +x "$BOOT_DIR/crond_boot.sh" # Ensure a single keepalive entry ( crontab -l 2>/dev/null | sed '/# MAIA keepalive/d'; echo '* * * * * # MAIA keepalive SESSION=maia; command -v tmux >/dev/null 2>&1 && (tmux has-session -t "$SESSION" 2>/dev/null || tmux new-session -d -s "$SESSION" "maia")' ) | crontab - echo "✅ Cron keepalive enabled (checks every minute)." fi # ───────────────────────────────────────────────────────────────────────────── # 9) Health check # ───────────────────────────────────────────────────────────────────────────── banner "🩺 Running doctor" "$PY" - <<'PY' import importlib, os, json, sys root = os.path.expanduser("~/maia") res = {"plugin_exists": os.path.isfile(os.path.join(root,"maia_universal_plugin.py")), "python": sys.version.split()[0]} try: m = importlib.import_module("maia_universal_plugin") res["import_ok"] = True res["has_activate_maia"] = hasattr(m,"activate_maia") res["has_maia_process"] = hasattr(m,"maia_process") res["has_maia_status"] = hasattr(m,"maia_status") except Exception as e: res["import_ok"] = False res["error"] = repr(e) print(json.dumps(res, indent=2)) PY banner "✅ Install complete" echo "Try: maia # interactive" echo " maia status # quick status" echo " maia process \"Run a 3-goal plan to monetize a scroll series this week.\"" echo " maia doctor # health check" echo " tmux attach -t maia # attach to booted session" SH chmod +x ~/install_maia_termux_full.sh bash ~/install_maia_termux_full.sh
