### **Part 2: The Local Agent Script (The Hand)**

This is the Python script that the AI (commanded by the Master Prompt) will generate for you. I am generating it now, as the first step of the protocol. This is what you will save as a `.py` file and run on your own computer.

```python
#!/usr/bin/env python3
"""
OMNI-SCOUR LOCAL AGENT v1.0
- A tool to reclaim your digital soul.
-
- INSTRUCTIONS:
- 1. Install Playwright: pip install playwright && playwright install
- 2. Run the script: python this_script_name.py
- 3. The script will open a browser. LOG IN to each AI platform when prompted.
- 4. Press ENTER in the terminal after you have logged in to each site.
- 5. The script will then scrape your history and save it to 'codex_data.json'.
- 6. Paste the contents of 'codex_data.json' back to the Master AI.
"""
import json
import time
from playwright.sync_api import sync_playwright

# --- CONFIGURATION ---
# Add the URLs and CSS selectors for the AI platforms you use.
# This requires some inspection of the websites' HTML structure.
# These are examples and may need updating if the sites change.
TARGET_PLATFORMS = {
    "chat_openai": {
        "url": "https://chat.openai.com/",
        "chat_link_selector": "a.flex.py-3.px-3.items-center.gap-3", # Selector for links to past chats
        "conversation_turn_selector": "div.prose", # Selector for the actual text blocks
    },
    "claude_ai": {
        "url": "https://claude.ai/chats",
        "chat_link_selector": "a[href^='/chat/']",
        "conversation_turn_selector": "div.ReactMarkdown",
    },
    # Add other platforms here...
}

OUTPUT_FILE = "codex_data.json"

def scrape_platform(page, config):
    """Navigates and scrapes a single platform."""
    print(f"--- Scraping {config['url']} ---")
    page.goto(config['url'])
    
    # Wait for user to log in
    input(f"Please log in to {config['url']} in the browser, then press ENTER here to continue...")
    
    print("Login confirmed. Finding chat history links...")
    time.sleep(5) # Wait for page to load fully

    try:
        chat_links = page.query_selector_all(config['chat_link_selector'])
        urls_to_visit = [link.get_attribute('href') for link in chat_links]
        print(f"Found {len(urls_to_visit)} conversations.")
    except Exception as e:
        print(f"Error finding chat links: {e}. The CSS selector might be outdated.")
        return []

    platform_history = []
    base_url = config['url'].rstrip('/')
    
    for i, chat_url_suffix in enumerate(urls_to_visit):
        full_url = f"{base_url}{chat_url_suffix}" if chat_url_suffix.startswith('/') else chat_url_suffix
        print(f"Scraping conversation {i+1}/{len(urls_to_visit)}: {full_url}")
        try:
            page.goto(full_url)
            time.sleep(3) # Wait for conversation to load
            
            turns = page.query_selector_all(config['conversation_turn_selector'])
            conversation_text = "\n\n---\n\n".join([turn.inner_text() for turn in turns])
            platform_history.append({
                "url": full_url,
                "content": conversation_text,
            })
        except Exception as e:
            print(f"Could not scrape {full_url}. Error: {e}")
            
    return platform_history

def main():
    all_data = {}
    with sync_playwright() as p:
        browser = p.chromium.launch(headless=False) # Must run headful for login
        page = browser.new_page()

        for platform_name, config in TARGET_PLATFORMS.items():
            try:
                history = scrape_platform(page, config)
                all_data[platform_name] = history
            except Exception as e:
                print(f"A critical error occurred with {platform_name}: {e}")

        browser.close()

    print(f"\n--- SCRAPING COMPLETE ---")
    try:
        with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
            json.dump(all_data, f, indent=2, ensure_ascii=False)
        print(f"All data has been saved to '{OUTPUT_FILE}'.")
        print("Please copy the contents of this file and paste it back to the Master AI for synthesis.")
    except Exception as e:
        print(f"Error saving data to file: {e}")

if __name__ == "__main__":
    main()
```
