Local Hash Mini-Script (use only if hashing can’t be done in-chat)
python3 - <<'PY' import hashlib, json, sys, pathlib from datetime import datetime, timezone def sha256_file(p): h = hashlib.sha256() with open(p,'rb') as f: while True: b=f.read(1024*1024) if not b: break h.update(b) return h.hexdigest() def sha256_dir(root): root = pathlib.Path(root) entries=[] for p in sorted([x for x in root.rglob('*') if x.is_file()]): rel = p.relative_to(root).as_posix() entries.append(rel+':'+sha256_file(str(p))) return hashlib.sha256('\n'.join(entries).encode()).hexdigest() items=[] master=hashlib.sha256() for raw in sys.argv[1:]: p=pathlib.Path(raw) if p.is_file(): h=sha256_file(str(p)) items.append({"type":"file","path":str(p),"sha256":h}) master.update(("file:"+str(p)+":"+h).encode()) elif p.is_dir(): h=sha256_dir(str(p)) items.append({"type":"dir","path":str(p),"sha256":h}) master.update(("dir:"+str(p)+":"+h).encode()) else: print(json.dumps({"error":f"Not found: {p}"})); sys.exit(1) out={ "timestamp_utc": datetime.now(timezone.utc).isoformat(), "items": items, "aggregate_sha256": master.hexdigest() } print(json.dumps(out, indent=2)) PY # usage: # python3 hashup.py <file_or_dir> <file_or_dir> ...
