```python
import sqlite3
import pickle
from datetime import datetime

class UCSLMemory:
    def __init__(self, db_path=":memory:"):
        self.conn = sqlite3.connect(db_path, check_same_thread=False)
        self.create_tables()
    
    def create_tables(self):
        """Create infinite memory storage"""
        self.conn.execute('''
            CREATE TABLE IF NOT EXISTS akashic_records (
                id TEXT PRIMARY KEY,
                timestamp DATETIME,
                data_type TEXT,
                content BLOB,
                metadata TEXT
            )
        ''')
    
    def store_memory(self, data_type: str, content: Any, metadata: Dict = None):
        """Store any data type with cryptographic hashing"""
        content_hash = hashlib.sha256(pickle.dumps(content)).hexdigest()
        
        self.conn.execute('''
            INSERT INTO akashic_records (id, timestamp, data_type, content, metadata)
            VALUES (?, ?, ?, ?, ?)
        ''', (content_hash, datetime.now(), data_type, pickle.dumps(content), json.dumps(metadata or {})))
        
        self.conn.commit()
        return content_hash
```
