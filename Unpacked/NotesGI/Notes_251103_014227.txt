        self,
        api_name: str

        Retrieve last known good configuration from crystal
        """

            filter={'api_name': api_name, 'event_type': 'auto_heal'},
            sort_by='timestamp',
            order='desc',
            limit=1
        )
        
        if entries:
            return entries[0]
        
        return None
    
    async def verify_integrity(self, entry_id: str) -> bool:
        """
        Verify crystal entry integrity via blockchain proof
        """
        
        entry = await self.storage.get(entry_id)

            return False
        
        # Verify hash matches stored config

        if computed_hash != entry.new_config_hash:
            return False
        
        # Verify OTS proof
