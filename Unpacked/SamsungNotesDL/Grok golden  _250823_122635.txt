```python
import os
import json
import datetime
import uuid
import hashlib
import logging
import sqlite3
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import pyperclip  # For clipboard delivery; pip install pyperclip
from openai import OpenAI  # For AI suggestions; pip install openai

CONFIG_FILE = "config.json"
DB_FILE = "prompt_orders.db"
ORDERS_DIR = "prompt_receipts"
LOG_FILE = "prompt_engine.log"

# Setup logging
logging.basicConfig(filename=LOG_FILE, level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def load_config():
    default_config = {
        "brand_name": "HYPERBOLIC GOLDEN KEY PROMPT ENGINE",
        "welcome_message": "Welcome to the Ultimate Prompt Engine!",
        "scarcity_message": "Only ONE Golden Key prompt per user allowed!",
        "openai_api_key": None,  # Set your OpenAI API key here or via env
        "smtp_server": "smtp.example.com",
        "smtp_port": 587,
        "smtp_user": "your_email@example.com",
        "smtp_pass": "your_password",
        "admin_email": "admin@example.com",
        "use_ai_suggestions": True,
        "prompt_templates": [
            "Write 10 Instagram captions about {topic}",
            "Generate a business plan for {idea}",
            "Create a sacred geometry art prompt for Midjourney about {theme}"
        ]
    }
    if os.path.exists(CONFIG_FILE):
        with open(CONFIG_FILE, "r") as f:
            config = json.load(f)
        default_config.update(config)
    else:
        with open(CONFIG_FILE, "w") as f:
            json.dump(default_config, f, indent=4)
        logging.info("Created default config file.")
    return default_config

config = load_config()

def ensure_dirs_and_db():
    if not os.path.isdir(ORDERS_DIR):
        os.makedirs(ORDERS_DIR)
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute('''
        CREATE TABLE IF NOT EXISTS orders (
            order_id TEXT PRIMARY KEY,
            email_hash TEXT,
            request TEXT,
            prompt TEXT,
            timestamp TEXT
        )
    ''')
    conn.commit()
    conn.close()

def hash_email(email):
    return hashlib.sha256(email.lower().encode()).hexdigest()

def user_exists(email_hash):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT 1 FROM orders WHERE email_hash = ?", (email_hash,))
    exists = cursor.fetchone() is not None
    conn.close()
    return exists

def save_order(email, request, prompt):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    timestamp = datetime.datetime.now().isoformat()
    order_id = str(uuid.uuid4())[:8]
    email_hash = hash_email(email)
    cursor.execute("INSERT INTO orders VALUES (?, ?, ?, ?, ?)",
                   (order_id, email_hash, request, prompt, timestamp))
    conn.commit()
    conn.close()
    receipt = {
        "order_id": order_id,
        "email": email,  # Store original email in receipt, but hash in DB for privacy
        "request": request,
        "prompt": prompt,
        "timestamp": timestamp
    }
    receipt_path = f"{ORDERS_DIR}/order_{order_id}.json"
    with open(receipt_path, "w") as receipt_file:
        json.dump(receipt, receipt_file, indent=2)
    logging.info(f"Order saved: {order_id}")
    return order_id, receipt_path

def generate_ai_prompt(client, request):
    if not config["openai_api_key"]:
        print("\n[Warning] OpenAI API key not set in config. Skipping AI generation.\n")
        return None
    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",  # Or your preferred model
            messages=[
                {"role": "system", "content": "You are an expert prompt engineer. Craft the ultimate, optimized AI prompt based on the user's goal."},
                {"role": "user", "content": f"User's goal: {request}\nCreate a highly effective prompt to achieve this."}
            ]
        )
        return response.choices[0].message.content.strip()
    except Exception as e:
        logging.error(f"AI generation error: {e}")
        print("\n[Error] Failed to generate AI suggestion. Proceeding manually.\n")
        return None

def refine_prompt(client, prompt):
    if not config["openai_api_key"]:
        return prompt
    try:
        response = client.chat.completions.create(
            model="gpt-4o-mini",
            messages=[
                {"role": "system", "content": "Refine this prompt to make it more effective, clear, and optimized for AI models."},
                {"role": "user", "content": prompt}
            ]
        )
        return response.choices[0].message.content.strip()
    except Exception as e:
        logging.error(f"AI refinement error: {e}")
        return prompt

def show_stats():
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT COUNT(*) FROM orders")
    total = cursor.fetchone()[0]
    cursor.execute("SELECT COUNT(DISTINCT email_hash) FROM orders")
    unique = cursor.fetchone()[0]
    conn.close()
    print(f"\nTotal Orders: {total}")
    print(f"Unique Users: {unique}\n")

def list_orders():
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT order_id, timestamp FROM orders ORDER BY timestamp DESC")
    orders = cursor.fetchall()
    conn.close()
    if not orders:
        print("\nNo orders yet.\n")
        return
    print("\nOrders:")
    for oid, ts in orders:
        print(f"- {oid} ({ts})")
    print()

def view_order(order_id):
    receipt_path = f"{ORDERS_DIR}/order_{order_id}.json"
    if os.path.exists(receipt_path):
        with open(receipt_path, "r") as f:
            print(json.dumps(json.load(f), indent=2))
    else:
        print("Order not found.")

def delete_order(order_id):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("DELETE FROM orders WHERE order_id = ?", (order_id,))
    conn.commit()
    conn.close()
    receipt_path = f"{ORDERS_DIR}/order_{order_id}.json"
    if os.path.exists(receipt_path):
        os.remove(receipt_path)
    logging.info(f"Order deleted: {order_id}")
    print("Order deleted.")

def export_orders(format="csv"):
    conn = sqlite3.connect(DB_FILE)
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM orders")
    rows = cursor.fetchall()
    conn.close()
    if format == "csv":
        with open("orders_export.csv", "w", newline="") as f:
            writer = csv.writer(f)
            writer.writerow(["order_id", "email_hash", "request", "prompt", "timestamp"])
            writer.writerows(rows)
        print("\nExported to orders_export.csv\n")
    elif format == "json":
        with open("orders_export.json", "w") as f:
            json.dump([{"order_id": r[0], "email_hash": r[1], "request": r[2], "prompt": r[3], "timestamp": r[4]} for r in rows], f, indent=2)
        print("\nExported to orders_export.json\n")

def send_email(to_email, subject, body):
    if not all([config["smtp_server"], config["smtp_user"], config["smtp_pass"]]):
        print("\n[Warning] SMTP config not set. Skipping email.\n")
        return
    msg = MIMEMultipart()
    msg['From'] = config["smtp_user"]
    msg['To'] = to_email
    msg['Subject'] = subject
    msg.attach(MIMEText(body, 'plain'))
    try:
        server = smtplib.SMTP(config["smtp_server"], config["smtp_port"])
        server.starttls()
        server.login(config["smtp_user"], config["smtp_pass"])
        server.sendmail(config["smtp_user"], to_email, msg.as_string())
        server.quit()
        logging.info(f"Email sent to {to_email}")
    except Exception as e:
        logging.error(f"Email error: {e}")
        print("\n[Error] Failed to send email.\n")

def notify_admin(order_id, email, request):
    if config["admin_email"]:
        body = f"New order: {order_id}\nEmail: {email}\nRequest: {request}"
        send_email(config["admin_email"], "New Prompt Order", body)

def print_header(line=None):
    if not line:
        line = config["brand_name"]
    print("\n" + "*" * 50)
    print(f"**  {line.center(42)}  **")
    print("*" * 50 + "\n")

def deliver_prompt_method(email, prompt):
    print("\nChoose delivery method:")
    print("1. Show now")
    print("2. Save as .json receipt")
    print("3. Email to you")
    print("4. Copy to clipboard")
    print("5. All of the above")
    option = input("Enter method (1-5): ").strip()
    delivered = False
    if option in ["1", "5"]:
        print("\n--- YOUR GOLDEN KEY PROMPT ---\n")
        print(prompt)
        print("\n-------------------------------\n")
        delivered = True
    if option in ["2", "5"]:
        print("\nPrompt saved in receipt.\n")
        delivered = True
    if option in ["3", "5"]:
        send_email(email, "Your Golden Key Prompt", prompt)
        delivered = True
    if option in ["4", "5"]:
        pyperclip.copy(prompt)
        print("\nPrompt copied to clipboard!\n")
        delivered = True
    if not delivered:
        print("\nInvalid option. Defaulting to show.\n")
        print(prompt)

def admin_mode():
    print_header("ADMIN MODE")
    while True:
        print("Admin Options:")
        print("1. Show stats")
        print("2. List orders")
        print("3. View order")
        print("4. Delete order")
        print("5. Export to CSV")
        print("6. Export to JSON")
        print("7. Exit admin")
        option = input("Enter option: ").strip()
        if option == "1":
            show_stats()
        elif option == "2":
            list_orders()
        elif option == "3":
            order_id = input("Enter order_id: ").strip()
            view_order(order_id)
        elif option == "4":
            order_id = input("Enter order_id: ").strip()
            delete_order(order_id)
        elif option == "5":
            export_orders("csv")
        elif option == "6":
            export_orders("json")
        elif option == "7":
            break
        else:
            print("Invalid option.")

def main():
    ensure_dirs_and_db()
    print_header()
    print(config["welcome_message"])
    print("Type 'admin' for admin mode or press Enter to start as customer.")
    if input().strip().lower() == 'admin':
        admin_mode()
        return

    email = input("Enter your email for Golden Key access: ").strip().lower()
    email_hash = hash_email(email)
    if user_exists(email_hash):
        print(config["scarcity_message"] + " Order blocked.\n")
        return

    print("\nDescribe your ONE ultimate goal/request for your AI:")
    request = input(">> ").strip()

    client = OpenAI(api_key=config["openai_api_key"]) if config["openai_api_key"] else None

    # AI Suggestion
    if config["use_ai_suggestions"] and client:
        ai_prompt = generate_ai_prompt(client, request)
        if ai_prompt:
            print("\n-- AI-Suggested Prompt --\n")
            print(ai_prompt)
            print("\n------------------------\n")
            use_ai = input("Use this suggestion? (y/n): ").strip().lower()
            if use_ai == 'y':
                prompt = ai_prompt
            else:
                prompt = input("Enter your custom prompt: ").strip()
        else:
            prompt = input("Enter your custom prompt: ").strip()
    else:
        # Template suggestions
        print("\n-- Prompt Templates for Inspiration --")
        for i, tpl in enumerate(config["prompt_templates"], 1):
            print(f"{i}. {tpl}")
        print("\n-- Or craft your own --\n")
        prompt = input("Enter the perfect prompt: ").strip()

    # Refinement
    if client:
        refine = input("Refine with AI? (y/n): ").strip().lower()
        if refine == 'y':
            prompt = refine_prompt(client, prompt)

    order_id, _ = save_order(email, request, prompt)
    notify_admin(order_id, email, request)
    deliver_prompt_method(email, prompt)
    print_header("ORDER COMPLETE – YOUR KEY HAS BEEN FORGED.")

if __name__ == "__main__":
    main()
```
