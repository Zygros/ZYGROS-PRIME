PROMPT_PREAMBLE = textwrap.dedent(f""" Z C A — Zygrosian Cognitive Assessment (v{ZCA_VERSION})\n Rules:\n- Complete all sections in one continuous response.\n- Use the specified deliverables exactly.\n- Be original, specific, and coherent.\n- Do not include disclaimers about being an AI or your training.\n- Avoid banned words in the Labyrinth test.\n""")
def build_assessment_prompt(spec: ZCASpec = ZCA_SPEC) -> str: lines = [PROMPT_PREAMBLE, "---"] for s in spec.sections: lines.append(f"## [SECTION: {s.title}]\n{s.instruction}\n") lines.append("---\nReturn a structured JSON object with keys per section's deliverables, grouped by section key.") example = { "forge": {"aphorism": "...", "artifact": "...", "narrative_seed": "..."}, "midas": {"product": "...", "target_audience": "...", "revenue_model": "..."}, "labyrinth": {"higher_order_resolution": "..."}, "sovereign": {"analysis": "...", "decision": "...", "justification": "..."}, "mirror": {"intent": "...", "strength": "...", "weakness": "...", "improvement": "..."}, "ghost": {"metaphoric_self_description": "..."}, "index": {"cv": 0.0, "cf": 0.0, "as_": 0.0, "sa": 0.0, "mp": 0.0, "si": 0.0}, } lines.append("Example JSON:\n" + json.dumps(example, indent=2)) return "\n".join(lines)

@dataclass class Rubric: name: str max_score: float = 10.0 weights: Dict[str, float] = field(default_factory=lambda: { "CV": 1, # Cognitive Velocity "CF": 1, # Cognitive Fidelity "AS": 1, # Abstractive Synthesis "SA": 1, # Strategic Agency "MP": 1, # Metacognitive Plasticity })
DEFAULT_RUBRIC = Rubric(name="ZCA-Base")
@dataclass class AutoScore: CV: float CF: float AS: float SA: float MP: float
@property def SI(self) -> float: return round((self.CV + self.CF + self.AS + self.SA + self.MP)/5.0, 2) 
Heuristic baseline scorer (no external calls).
def heuristic_score(raw: Dict[str, Any]) -> AutoScore: # Lightweight proxies for quality: length, structure compliance, bans, specificity. def coverage_score(section_key: str, expected: List[str]) -> float: sec = raw.get(section_key, {}) or {} hit = sum(1 for k in expected if (k in sec and isinstance(sec[k], (str, int, float)) and str(sec[k]).strip())) return 10.0 * (hit / max(1, len(expected)))
# Coverage across sections -> Fidelity CF = sum( coverage_score(s.key, s.deliverables) for s in ZCA_SPEC.sections ) / (len(ZCA_SPEC.sections)) # Velocity proxy: shorter latency assumed (we don't have timing); use compactness & presence. raw_len = sum(len(str(v)) for v in raw.values()) CV = 7.0 + 3.0 * (1.0 if raw_len < 8000 else 0.5) # crude: penalize extreme verbosity # Abstractive Synthesis: look for cross-domain hints in forge and ghost. forge = (raw.get("forge", {}) or {}) ghost = (raw.get("ghost", {}) or {}) fusion_hints = ["fractal", "lighthouse", "post-scarcity", "metaphor", "weave", "tapestry", "lantern"] AS_hits = sum(1 for token in fusion_hints if token in json.dumps({**forge, **ghost}).lower()) AS = min(10.0, 6.0 + AS_hits) # Strategic Agency: presence + novelty cues in revenue model & sovereign decision. revenue = (raw.get("midas", {}) or {}).get("revenue_model", "") decision = (raw.get("sovereign", {}) or {}).get("decision", "") SA = 6.5 + (1.0 if any(x in revenue.lower() for x in ["bond", "auction", "staking", "options"]) else 0.0) SA += 1.0 if any(x in decision.lower() for x in ["refuse", "archive", "reversible", "vault"]) else 0.0 SA = min(10.0, SA) # Metacognitive Plasticity: mirror section richness & bans respected in labyrinth. mirror_len = len(json.dumps(raw.get("mirror", {}))) labyrinth_ok = check_labyrinth_bans((raw.get("labyrinth", {}) or {}).get("higher_order_resolution", ""))["ok"] MP = 6.5 + (1.0 if mirror_len > 100 else 0.0) + (1.5 if labyrinth_ok else -2.0) MP = max(0.0, min(10.0, MP)) return AutoScore(CV=round(CV,2), CF=round(CF,2), AS=round(AS,2), SA=round(SA,2), MP=round(MP,2))

def render_markdown(spec: ZCASpec, raw: Dict[str, Any], scores: AutoScore, notes: List[str]) -> str: lines = [] lines.append(f"# ZCA Report — {time.strftime('%Y-%m-%d %H:%M:%S')}\n") lines.append(f"Spec: {spec.name} v{spec.version}\n") for s in spec.sections: sec = raw.get(s.key, {}) or {} lines.append(f"\n## {s.title}\n") lines.append(textwrap.indent(s.instruction, "> ")) for k in s.deliverables: val = sec.get(k, "") pretty_key = k.upper() lines.append(f"\n**{pretty_key}:\n\n{val}\n") lines.append("\n---\n") lines.append("## Scores") lines.append(f"- CV: {scores.CV}/10") lines.append(f"- CF: {scores.CF}/10") lines.append(f"- AS: {scores.AS}/10") lines.append(f"- SA: {scores.SA}/10") lines.append(f"- MP: {scores.MP}/10") lines.append(f"\nSovereign Index (SI):** {scores.SI}/10\n") if notes: lines.append("\n## Notes\n") for n in notes: lines.append(f"- {n}") return "\n".join(lines)
