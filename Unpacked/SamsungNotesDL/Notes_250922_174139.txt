import os
import json
import time
from datetime import datetime
from selenium import webdriver
from selenium.webdriver.chrome.service import Service
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from webdriver_manager.chrome import ChromeDriverManager

# --- CONFIGURATION ---
OUTPUT_DIR = "chronicler_output"
os.makedirs(OUTPUT_DIR, exist_ok=True)

# --- BROWSER AUTOMATION CORE ---
def setup_driver():
    """Initializes a Chrome WebDriver instance."""
    print("[INFO] Initializing Chrome WebDriver...")
    service = Service(ChromeDriverManager().install())
    options = webdriver.ChromeOptions()
    # Add options to make detection harder if needed, but start simple.
    # options.add_argument("--disable-blink-features=AutomationControlled")
    driver = webdriver.Chrome(service=service, options=options)
    return driver

def scrape_chatgpt(driver):
    """Navigates to ChatGPT and scrapes all conversation histories."""
    print("\n[CHRONICLER] Navigating to ChatGPT...")
    driver.get("https://chat.openai.com/")
    
    input("Please log in to your OpenAI account in the browser window, then press Enter here to continue...")
    
    print("[CHRONICLER] Login confirmed. Starting scrape...")
    
    # This selector targets the links to individual conversations in the sidebar.
    # NOTE: This is fragile and will break if OpenAI changes their site structure.
    conversation_links_selector = "a.flex.p-3.items-center"
    
    try:
        WebDriverWait(driver, 30).until(
            EC.presence_of_element_located((By.CSS_SELECTOR, conversation_links_selector))
        )
        print("[CHRONICLER] Found conversation list.")
        
        # Get all conversation links
        conv_elements = driver.find_elements(By.CSS_SELECTOR, conversation_links_selector)
        conv_urls = [elem.get_attribute('href') for elem in conv_elements]
        
        print(f"[INFO] Found {len(conv_urls)} conversations to archive.")
        
        all_chats = []
        for i, url in enumerate(conv_urls):
            print(f"  -> Scraping conversation {i+1}/{len(conv_urls)}...")
            driver.get(url)
            time.sleep(3) # Wait for content to load

            # This selector targets the main conversation content area.
            # It's designed to grab the whole thread.
            chat_container_selector = "div.text-base"
            
            WebDriverWait(driver, 20).until(
                EC.presence_of_element_located((By.CSS_SELECTOR, chat_container_selector))
            )
            
            messages = driver.find_elements(By.CSS_SELECTOR, chat_container_selector)
            conversation_text = "\n\n---\n\n".join([msg.text for msg in messages])
            
            chat_data = {
                "source": "ChatGPT",
                "url": url,
                "scraped_at": datetime.utcnow().isoformat(),
                "content": conversation_text
            }
            all_chats.append(chat_data)

        # Save to a single file
        filename = os.path.join(OUTPUT_DIR, f"chatgpt_archive_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}.json")
        with open(filename, 'w', encoding='utf-8') as f:
            json.dump(all_chats, f, indent=2, ensure_ascii=False)
            
        print(f"[SUCCESS] ChatGPT archive saved to {filename}")

    except Exception as e:
        print(f"[ERROR] Failed to scrape ChatGPT: {e}")
        print("[INFO] Please ensure you are logged in and the page structure has not changed significantly.")
