from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import torch
import torch.nn as nn
import torch.optim as optim
import torch.nn.functional as F
from torch.distributions import Categorical
import numpy as np
import hashlib  # For truth-hash provenance

app = FastAPI(title="Truth-Rigor Engine")

class Query(BaseModel):
    input: str
    iterations: int = 3  # Rapid RL loops

class TruthOutput(BaseModel):
    refined: str
    truth_score: float  # 0-1 curiosity/rigor
    audit: dict  # Bias flags, sources

# Simple Actor-Critic model for RL text refinement (feedforward for simplicity)
class ActorCritic(nn.Module):
    def __init__(self, input_size=64, hidden_size=32, output_size=10):  # Output for mock 'refinement actions'
        super(ActorCritic, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.actor = nn.Linear(hidden_size, output_size)
        self.critic = nn.Linear(hidden_size, 1)

    def forward(self, x):
        # x is a 1D vector (e.g., hashed query embedding)
        x = F.relu(self.fc1(x))
        actor_logits = self.actor(x)
        value = self.critic(x)
        return actor_logits, value

# Mock embedding: Hash query to vector
def query_to_vector(query: str, dim: int = 64):
    # Simple hash-based embedding
    hash_val = int(hashlib.sha256(query.encode()).hexdigest(), 16) % (10**8)
    return torch.tensor(np.random.RandomState(hash_val).random_sample(dim), dtype=torch.float32).unsqueeze(0)

# Global model and optimizer (shared across calls for incremental learning)
model = ActorCritic()
optimizer = optim.Adam(model.parameters(), lr=0.001)

def rl_truth_iterate_real(query: str, iters: int) -> tuple[str, float]:
    current = query
    total_reward = 0.0
    q_vec = query_to_vector(query)
    for _ in range(iters):
        logits, value = model(q_vec)
        probs = F.softmax(logits, dim=-1)
        dist = Categorical(probs)
        action = dist.sample()
        # Mock reward: higher for 'curious' actions (e.g., action >5)
        reward = 1.0 if action.item() > 5 else 0.0
        total_reward += reward
        # Mock refinement based on action
        refinements = [" – Maximally curious?", " – Truth refined", " – Aligned intent", " – Cosmic bridge", " – Soul infused"]
        current = current + refinements[action.item() % len(refinements)]
        # Simple policy loss
        log_prob = dist.log_prob(action)
        advantage = reward - value.item()
        loss = -(log_prob * advantage).mean()
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    score = min(total_reward / iters, 1.0)  # Cap at 1.0
    return current, score

@app.post("/forge_truth", response_model=TruthOutput)
async def forge_truth(q: Query):
    refined, score = rl_truth_iterate_real(q.input, q.iterations)
    audit_hash = hashlib.sha256(refined.encode()).hexdigest()[:8]
    return TruthOutput(
        refined=refined,
        truth_score=score,
        audit={"provenance": audit_hash, "flags": {"bias_detected": score < 0.7}}
    )

if __name__ == "__main__":
