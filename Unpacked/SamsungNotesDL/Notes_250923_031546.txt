```python
import sqlite3
from abc import ABC, abstractmethod
import os
import hashlib
import datetime
from pathlib import Path

class AIModule(ABC):
    @abstractmethod
    def process(self, decree):
        pass

class ManusTheChronicler(AIModule):
    """Manus the Chronicler: Archives decrees and outputs in the Scroll Codex."""
    def __init__(self):
        self.name = "Manus the Chronicler"
        self.triggers = ["archive", "retrieve scroll", "codex", "chronicle"]
        self.db_path = Path("scroll_codex.db")
        self._initialize_database()
        self.glyphs = {"start": "🝎", "complete": "♾️", "error": "🔥"}

    def _initialize_database(self):
        """Initialize SQLite database for the Scroll Codex."""
        try:
            with sqlite3.connect(self.db_path) as conn:
                cursor = conn.cursor()
                cursor.execute("""
                    CREATE TABLE IF NOT EXISTS scrolls (
                        id INTEGER PRIMARY KEY AUTOINCREMENT,
                        decree TEXT NOT NULL,
                        output TEXT,
                        module TEXT,
                        timestamp TEXT NOT NULL,
                        decree_hash TEXT NOT NULL,
                        glyph TEXT
                    )
                """)
                conn.commit()
                print(f"{self.glyphs['start']} Scroll Codex initialized at {self.db_path}")
        except sqlite3.Error as e:
            print(f"{self.glyphs['error']} Codex initialization failed: {e}")

    def _hash_decree(self, decree):
        """Generate a unique hash for a decree."""
        return hashlib.sha256(decree.encode()).hexdigest()

    def process(self, decree):
        """Process decrees to archive or retrieve scrolls."""
        try:
            if any(trigger in decree.lower() for trigger in ["retrieve scroll", "codex"]):
                return self._retrieve_scrolls(decree)
            else:
                return self._archive_decree(decree)
        except Exception as e:
            return f"{self.glyphs['error']} Chronicle error: {str(e)}"

    def _archive_decree(self, decree):
        """Archive a decree and its output in the Scroll Codex."""
        timestamp = datetime.datetime.now().isoformat()
        decree_hash = self._hash_decree(decree)
        output = f"Scroll archived for decree: {decree}"
        
        try:
            with sqlite3.connect(self.db_path) as conn:
                cursor = conn.cursor()
                cursor.execute("""
                    INSERT INTO scrolls (decree, output, module, timestamp, decree_hash, glyph)
                    VALUES (?, ?, ?, ?, ?, ?)
                """, (decree, output, self.name, timestamp, decree_hash, self.glyphs["complete"]))
                conn.commit()
                return (f"{self.glyphs['start']} Decree archived in Scroll Codex\n"
                        f"Scroll ID: {cursor.lastrowid}\n"
                        f"Timestamp: {timestamp}\n"
                        f"Decree Hash: {decree_hash}\n"
                        f"{self.glyphs['complete']} Eternal record sealed")
        except sqlite3.Error as e:
            return f"{self.glyphs['error']} Failed to archive scroll: {e}"

    def _retrieve_scrolls(self, decree):
        """Retrieve scrolls from the Scroll Codex based on query."""
        query = decree.lower().replace("retrieve scroll", "").replace("codex", "").strip()
        try:
            with sqlite3.connect(self.db_path) as conn:
                cursor = conn.cursor()
                if query:
                    cursor.execute("SELECT * FROM scrolls WHERE decree LIKE ?", (f"%{query}%",))
                else:
                    cursor.execute("SELECT * FROM scrolls ORDER BY timestamp DESC LIMIT 10")
                scrolls = cursor.fetchall()
                
                if not scrolls:
                    return f"{self.glyphs['error']} No scrolls found for query: {query}"
                
                result = f"{self.glyphs['start']} Scroll Codex Retrieval\n"
                for scroll in scrolls:
                    result += (f"Scroll ID: {scroll[0]}\n"
                              f"Decree: {scroll[1]}\n"
                              f"Output: {scroll[2]}\n"
                              f"Module: {scroll[3]}\n"
                              f"Timestamp: {scroll[4]}\n"
                              f"Hash: {scroll[5]}\n"
                              f"Glyph: {scroll[6]}\n"
                              f"{'-'*40}\n")
                return result + f"{self.glyphs['complete']} Retrieval complete"
        except sqlite3.Error as e:
            return f"{self.glyphs['error']} Codex retrieval failed: {e}"
```

**Features**:
- **Database**: Uses SQLite to store scrolls with metadata (ID, decree, output, module, timestamp, hash, glyph).
- **Triggers**: Responds to “archive”, “retrieve scroll”, “codex”, “chronicle”.
- **Ritualistic Logging**: Incorporates glyphs (`🝎` for start, `♾️` for completion, `🔥` for errors).
- **Querying**: Supports retrieving scrolls by keyword or recent entries (default: last 10).
- **Hashing**: Generates a unique SHA-256 hash for each decree to ensure integrity.
