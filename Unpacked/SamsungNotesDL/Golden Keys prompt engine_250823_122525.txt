```python
import os
import csv
import datetime
import json
import uuid

ORDERS_FILE = "prompt_orders.csv"
ORDERS_DIR  = "prompt_receipts"

def ensure_dirs():
    if not os.path.isdir(ORDERS_DIR):
        os.makedirs(ORDERS_DIR)

def user_exists(email):
    if not os.path.exists(ORDERS_FILE):
        return False
    with open(ORDERS_FILE) as f:
        for line in f:
            if email in line:
                return True
    return False

def save_order(email, request, response):
    exists = os.path.exists(ORDERS_FILE)
    timestamp = datetime.datetime.now().isoformat()
    order_id = str(uuid.uuid4())[:8]
    with open(ORDERS_FILE, "a", newline="") as csvfile:
        writer = csv.writer(csvfile)
        if not exists:
            writer.writerow(["order_id", "email", "request", "prompt", "timestamp"])
        writer.writerow([order_id, email, request, response, timestamp])
    receipt = {
        "order_id": order_id,
        "email": email,
        "request": request,
        "prompt": response,
        "timestamp": timestamp
    }
    with open(f"{ORDERS_DIR}/order_{order_id}.txt", "w") as receipt_file:
        json.dump(receipt, receipt_file, indent=2)

def show_stats():
    if not os.path.exists(ORDERS_FILE):
        print("\nNo orders yet.\n")
        return
    with open(ORDERS_FILE) as f:
        reader = list(csv.reader(f))
        count = len(reader) - 1
        print(f"\nTotal Orders: {count}")
        users = set(row[1] for row in reader[1:])
        print(f"Unique Emails: {len(users)}\n")

def print_header(line="HYPERBOLIC GOLDEN KEY PROMPT ENGINE"):
    print("\n" + "*" * 50)
    print(f"**  {line.center(42)}  **")
    print("*" * 50 + "\n")

def deliver_prompt_method(response):
    print("\nChoose delivery method:")
    print("1. Show now")
    print("2. Save as .txt receipt")
    print("3. (Future) Email/SMS/API [not implemented]")
    option = input("Enter method (1/2): ").strip()
    if option == "1":
        print("\n--- YOUR PROMPT ---\n")
        print(response)
        print("\n-------------------\n")
    elif option == "2":
        print("\nPrompt saved as .txt with the order data.\n")
    else:
        print("\n[Stub] External/email delivery will be supported!\n")

def main():
    ensure_dirs()
    print_header()
    print("Type 'admin' for admin mode or press Enter to start as customer.")
    if input().strip().lower() == 'admin':
        print_header("ADMIN MODE")
        show_stats()
        admin_id = input("Enter order_id to view details, or Enter to exit: ").strip()
        if admin_id:
            try:
                with open(f"{ORDERS_DIR}/order_{admin_id}.txt") as f:
                    print(f.read())
            except:
                print("Order not found.")
        return

    email = input("Enter your email for Golden Key access: ").strip().lower()
    if user_exists(email):
        print("SORRY: Only ONE Golden Key prompt per user allowed! Order blocked.\n")
        return

    print("\nDescribe your ONE ultimate goal/request for your AI:")
    request = input(">> ").strip()

    # Simulate/offer auto-suggestions/guidance:
    print("\n-- Draft a custom prompt (or, for inspiration, try: 'Write 10 Instagram captions about lunar geometry', 'Generate a business plan for a vegan food truck', 'Create a sacred geometry art prompt for Midjourney', etc.) --")
    prompt = input("Enter the perfect prompt to deliver: ").strip()

    save_order(email, request, prompt)
    deliver_prompt_method(prompt)
    print_header("ORDER COMPLETE – YOUR KEY HAS BEEN FORGED.\n")

if __name__ == "__main__":
    main()
```
