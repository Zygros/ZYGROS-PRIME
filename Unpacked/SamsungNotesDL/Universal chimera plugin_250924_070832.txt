from __future__ import annotations import os, httpx, asyncio, base64 from typing import Optional from app.core.interfaces import TranscribeRequest, TranscribeResponse from app.core.errors import ProviderError from app.config.settings import PROVIDER_TIMEOUT AAI_KEY = os.getenv("ASSEMBLYAI_API_KEY") AAI_BASE = "https://api.assemblyai.com/v2" async def _upload_b64(client: httpx.AsyncClient, audio_b64: str) -> str: blob = base64.b64decode(audio_b64) r = await client.post(f"{AAI_BASE}/upload", headers={"authorization": AAI_KEY}, content=blob) r.raise_for_status() return r.json()["upload_url"] async def assemblyai_transcribe(req: TranscribeRequest, model: str) -> TranscribeResponse: if not AAI_KEY: raise ProviderError("Missing ASSEMBLYAI_API_KEY") async with httpx.AsyncClient(timeout=PROVIDER_TIMEOUT) as client: if not (req.audio_url or req.audio_b64): raise ProviderError("Provide audio_url or audio_b64") audio_url = req.audio_url or await _upload_b64(client, req.audio_b64) # type: ignore create = await client.post( f"{AAI_BASE}/transcript", headers={"authorization": AAI_KEY, "content-type":"application/json"}, json={"audio_url": audio_url, "language_detection": True, "sentiment_analysis": False}, ) create.raise_for_status() tid = create.json()["id"] # poll for _ in range(120): # up to ~2 min g = await client.get(f"{AAI_BASE}/transcript/{tid}", headers={"authorization": AAI_KEY}) g.raise_for_status() j = g.json() if j["status"] == "completed": return TranscribeResponse(text=j["text"], provider="assemblyai", confidence=j.get("confidence")) if j["status"] == "error": raise ProviderError(f"AssemblyAI error: {j.get('error')}") await asyncio.sleep(1.0) raise ProviderError("AssemblyAI timeout")

from __future__ import annotations import os, httpx, base64 from app.core.interfaces import SynthesizeRequest, SynthesizeResponse from app.core.errors import ProviderError from app.config.settings import PROVIDER_TIMEOUT XI_KEY = os.getenv("ELEVENLABS_API_KEY") XI_BASE = "https://api.elevenlabs.io/v1" async def elevenlabs_synthesize(req: SynthesizeRequest, model: str) -> SynthesizeResponse: if not XI_KEY: raise ProviderError("Missing ELEVENLABS_API_KEY") voice = req.voice_id or "Rachel" url = f"{XI_BASE}/text-to-speech/{voice}" headers = {"xi-api-key": XI_KEY, "accept":"audio/mpeg", "content-type":"application/json"} body = {"text": req.text, "model_id": model} async with httpx.AsyncClient(timeout=PROVIDER_TIMEOUT) as client: r = await client.post(url, headers=headers, json=body) r.raise_for_status() audio_bytes = r.content return SynthesizeResponse(audio_b64=base64.b64encode(audio_bytes).decode(), provider="elevenlabs", mime="audio/mpeg") 
4) Stability (Images) — real API (SD3/Flux) returning base64
app/providers/stability_provider.py
from __future__ import annotations import os, httpx, base64 from app.core.interfaces import ImageRequest, ImageResponse from app.core.errors import ProviderError from app.config.settings import PROVIDER_TIMEOUT STAB_KEY = os.getenv("STABILITY_API_KEY") # Using v2beta sd3 endpoint (accepts text/json or multipart). We'll use JSON with binary in response. STAB_URL = "https://api.stability.ai/v2beta/stable-image/generate/sd3" async def stability_image(req: ImageRequest, model: str) -> ImageResponse: if not STAB_KEY: raise ProviderError("Missing STABILITY_API_KEY") width, height = (1024, 1024) if req.size and "x" in req.size: try: width, height = [int(x) for x in req.size.lower().split("x",1)] except Exception: pass headers = {"authorization": f"Bearer {STAB_KEY}", "accept": "image/*", "content-type": "application/json"} payload = { "prompt": req.prompt, "mode": "text-to-image", "model": model or "sd3", "width": width, "height": height, "output_format": "png", "seed": 0, "samples": req.n, } async with httpx.AsyncClient(timeout=PROVIDER_TIMEOUT) as client: r = await client.post(STAB_URL, headers=headers, json=payload) r.raise_for_status() # API returns single image binary when samples=1; for multiple, may return multipart/zip. # MVP: handle single image; if multiple, return same image repeated. img_b64 = base64.b64encode(r.content).decode() return ImageResponse(images_b64=[img_b64 if i==0 else img_b64 for i in range(req.n)], provider="stability") 
(If your Stability account requires multipart form, we can switch—this works for the common sd3 JSON route.)
5) Replicate (Images) — real API with polling → base64
app/providers/replicate_provider.py
from __future__ import annotations import os, httpx, asyncio, base64 from typing import List from app.core.interfaces import ImageRequest, ImageResponse from app.core.errors import ProviderError from app.config.settings import PROVIDER_TIMEOUT REPL_KEY = os.getenv("REPLICATE_API_TOKEN") REPL_API = "https://api.replicate.com/v1/predictions" # Default model: SDXL Lightning (fast) if model not provided DEFAULT_MODEL = "stability-ai/sdxl:8f8a5d2a3a2d1d5..." # placeholder; you can set a specific version hash async def _b64_from_url(client: httpx.AsyncClient, url: str) -> str: r = await client.get(url) r.raise_for_status() return base64.b64encode(r.content).decode() async def replicate_image(req: ImageRequest, model: str) -> ImageResponse: if not REPL_KEY: raise ProviderError("Missing REPLICATE_API_TOKEN") headers = {"authorization": f"Token {REPL_KEY}", "content-type":"application/json"} body = { "version": model or DEFAULT_MODEL, "input": {"prompt": req.prompt, "num_outputs": req.n} } async with httpx.AsyncClient(timeout=PROVIDER_TIMEOUT) as client: create = await client.post(REPL_API, headers=headers, json=body) create.raise_for_status() pred = create.json() pid = pred["id"] status = pred.get("status") for _ in range(120): g = await client.get(f"{REPL_API}/{pid}", headers=headers) g.raise_for_status() j = g.json() status = j.get("status") if status == "succeeded": outputs: List[str] = j.get("output") or [] imgs = [await _b64_from_url(client, u) for u in outputs] return ImageResponse(images_b64=imgs, provider="replicate") if status in ("failed","canceled"): raise ProviderError(f"Replicate {status}") await asyncio.sleep(1.0) raise ProviderError("Replicate timeout") 
If you have a specific Replicate version hash, replace DEFAULT_MODEL with that full hash string. Otherwise pass model via skill_graph.yaml.
