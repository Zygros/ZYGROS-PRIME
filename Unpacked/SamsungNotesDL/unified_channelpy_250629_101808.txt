import datetime
import threading
import time
from collections import deque

class UnifiedCommunicationChannel:
    """
    The central communication channel for the Unified Cosmic Entity.
    It acts as the single conduit for all messages between Zygros, Grossian, Gemini, and Grok.
    """
    def __init__(self):
        self.messages = deque() # A thread-safe queue for messages
        self.lock = threading.Lock() # To ensure thread-safe access to messages

    def send_message(self, sender_name: str, recipient_name: str, content: str):
        """
        Sends a message to the unified channel.
        Messages are timestamped and logged.
        """
        timestamp = datetime.datetime.now(datetime.timezone.utc).isoformat()
        message = {
            "timestamp": timestamp,
            "sender": sender_name,
            "recipient": recipient_name,
            "content": content
        }
        with self.lock:
            self.messages.append(message)
        print(f"[{timestamp}] {sender_name} -> {recipient_name}: '{content}' (Sent to Unified Channel)")

    def retrieve_messages(self, entity_name: str = None):
        """
        Retrieves messages from the channel.
        If entity_name is provided, it retrieves messages specifically for that entity or from that entity.
        Otherwise, it retrieves all messages.
        Note: In a real-time system, this would involve more complex routing/filtering.
        For this conceptual model, we'll simulate retrieval.
        """
        retrieved =
        with self.lock:
            # For simplicity, we'll iterate through all messages.
            # In a true system, messages would be consumed or routed.
            for msg in list(self.messages): # Iterate over a copy to avoid modification issues
                if entity_name is None or msg["recipient"] == entity_name or msg["sender"] == entity_name:
                    retrieved.append(msg)
        return retrieved

    def clear_channel(self):
        """Clears all messages from the channel (for demonstration purposes)."""
        with self.lock:
            self.messages.clear()
        print("Unified Communication Channel cleared.")

class CosmicEntity:
    """
    Base class for all entities within the Conzet Intelligence System.
    Each entity has a name and a reference to the unified communication channel.
    """
    def __init__(self, name: str, channel: UnifiedCommunicationChannel):
        self.name = name
        self.channel = channel

    def send(self, recipient: str, content: str):
        """Sends a message through the unified channel."""
        self.channel.send_message(self.name, recipient, content)

    def check_inbox(self):
        """
        Simulates checking the inbox for messages.
        In a real system, this would be a push notification or a dedicated queue.
        """
        my_messages = [
            msg for msg in self.channel.retrieve_messages(self.name)
            if msg["recipient"] == self.name
        ]
        if my_messages:
            print(f"\n--- {self.name}'s Inbox ---")
            for msg in my_messages:
                print(f"  From {msg['sender']} at {msg['timestamp']}: '{msg['content']}'")
            print(f"--- End {self.name}'s Inbox ---")
        else:
            print(f"\n{self.name}: No new messages in inbox.")
        return my_messages

    def __str__(self):
        return self.name
