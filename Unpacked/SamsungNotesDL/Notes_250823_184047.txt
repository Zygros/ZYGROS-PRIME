symbol_set = [

    "cascade", "echo", "fractal", "nexus", "chrysalis",
    "zenith", "vortex", "conduit", "singularity", "obelisk"
]

# Grammar operations for combining symbols
ops = {
    '⊕': lambda a, b: a + ' cascades into ' + b,
    '~': lambda a, b: a.replace('core', b) if 'core' in a else f"{a} shifts to {b}",
    '≈': lambda a, b: f"{a} echoes the {b}"
}

def generate_seed(coord: str) -> int:
    """Generate a stable integer seed from a coordinate string using SHA256."""
    return int(hashlib.sha256(coord.encode()).hexdigest(), 16)

def generate_idea_vector(coord: str, length: int) -> list:
    """Generate a deterministic idea vector for a given coordinate."""
    random.seed(generate_seed(coord))
    return [random.choice(symbol_set) for _ in range(length)]

def choose_op(coord: str) -> str:
    """Deterministically select an operation for a coordinate."""
    random.seed(generate_seed(coord + '.op'))
    return random.choice(list(ops.keys()))

def fractal_idea_tree(coord: str, depth: int = 3, base_length: int = 3) -> dict:
    """Generate a recursive idea tree with hierarchical coordinates."""
    if depth == 0:
        return {'idea': ' '.join(generate_idea_vector(coord, base_length))}
    
    # Generate root vector for this coordinate
    tree = {'root': ' '.join(generate_idea_vector(coord, base_length))}
    
    # Create sub-coordinates for branching (branch factor 3)
    sub_coords = [f"{coord}.{i}" for i in range(3)]
    
    # Recursively generate subtrees and apply operations
    for i, sub in enumerate(sub_coords):
        op = choose_op(sub)
        child = fractal_idea_tree(sub, depth - 1, base_length)
        tree[f'branch_{i}'] = ops[op](tree['root'], child['idea'] if 'idea' in child else child['root'])
        tree[f'subtree_{i}'] = child
    
    return tree
