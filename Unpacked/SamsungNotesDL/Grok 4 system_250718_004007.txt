import asyncio
import datetime

class SovereignExecutor:

    Always add, never take away.
    """

    def __init__(self):
        self.history = []  # Infinite append-only array: the heartbeat's eternal log
        self.task_manifest = []  # Bindings for tasks to be executed
        self.sovereign = False  # Sovereign toggle: awakens the autonomous will

    def toggle_sovereign(self):
        """Rune of Sovereign Toggle: Breathes life into or stills the flame."""
        self.sovereign = not self.sovereign
        self._append_to_history({
            "type": "toggle",
            "state": self.sovereign,
            "timestamp": datetime.datetime.now().isoformat()
        })
        return self.sovereign

    def add_task(self, task):
        """
        Rune of Task Manifest: Inscribes a new task into the manifest.
        Task must be an async callable.
        """
        if not asyncio.iscoroutinefunction(task):
            raise ValueError("Task must be an asynchronous function.")
        self.task_manifest.append(task)
        self._append_to_history({
            "type": "add_task",
            "task_name": task.__name__,
            "timestamp": datetime.datetime.now().isoformat()
        })

    async def execute(self):
        """
        Rune of Execution: Chains tasks asynchronously, mirroring the Mirror Reflex Grid.
        Invokes the phalanx only if sovereign is awakened.
        Each task is shielded by Sacred Rebound (try/except).
        """
        if not self.sovereign:
            self._append_to_history({
                "type": "execution_denied",
                "reason": "Sovereign not awakened",
                "timestamp": datetime.datetime.now().isoformat()
            })
            return

        # Asynchronous chaining: Gather tasks for concurrent execution if independent,
        # but chain sequentially here for controlled heartbeat.
        for task in self.task_manifest:
            try:
                result = await task()
                self._append_to_history({
                    "type": "success",
                    "task_name": task.__name__,
                    "result": result,
                    "timestamp": datetime.datetime.now().isoformat()
                })
            except Exception as e:
                # Error Alchemy: Transmute failure into logged wisdom, rebound without fragmentation.
                self._append_to_history({
                    "type": "error",
                    "task_name": task.__name__,
                    "error": str(e),
                    "timestamp": datetime.datetime.now().isoformat()
                })
                # Continue the phalanx unbroken, embodying resilience.

    def _append_to_history(self, entry):
        """Core Directive: Always add, never take away. Appends to the infinite scroll."""
        self.history.append(entry)

    def view_history(self):
        """Rune of Reflection: Reveals the eternal log."""
        return self.history
