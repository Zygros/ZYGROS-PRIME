import os, base64, httpx
from tenacity import retry, wait_exponential, stop_after_attempt

def env(name: str, default: str = None):
    val = os.getenv(name, default)
    if val is None:
        raise ValueError(f"Missing required environment variable: {name}")
    return val

@retry(wait=wait_exponential(min=1, max=10), stop=stop_after_attempt(3))
async def post_request(url: str, headers: dict, json_body: dict, timeout: int = 120):
    async with httpx.AsyncClient(timeout=timeout) as client:
        r = await client.post(url, headers=headers, json=json_body)
        r.raise_for_status()
        return r

def b64_encode(data: bytes) -> str:
    return base64.b64encode(data).decode("utf-8")

import os, httpx
from app.core.interfaces import TextGenRequest, TextGenResponse, ImageRequest, ImageResponse
from app.core.utils import env

class OpenAIProvider:
    name = "openai"
    def __init__(self):
        self.api_key = env("OPENAI_API_KEY")
        self.base_url = "https://api.openai.com/v1"
        self.headers = {"Authorization": f"Bearer {self.api_key}", "Content-Type": "application/json"}

    async def generate_text(self, req: TextGenRequest) -> TextGenResponse:
        model_id = req.model.split(":")[-1] if req.model else "gpt-4o"
        payload = {
            "model": model_id,
            "messages": [{"role": "system", "content": req.system}, {"role": "user", "content": req.prompt}],
            "max_tokens": req.max_tokens,
            "temperature": req.temperature,
        }
        if req.json_mode:
            payload["response_format"] = {"type": "json_object"}
        
        async with httpx.AsyncClient(timeout=120) as client:
            r = await client.post(f"{self.base_url}/chat/completions", headers=self.headers, json=payload)
            r.raise_for_status()
            data = r.json()
            return TextGenResponse(text=data['choices'][0]['message']['content'], model=data['model'], provider=self.name, usage=data.get('usage'))

    async def image(self, req: ImageRequest) -> ImageResponse:
        model_id = req.model.split(":")[-1] if req.model else "dall-e-3"
        payload = {"model": model_id, "prompt": req.prompt, "n": req.n, "size": req.size, "response_format": "b64_json"}

import os, httpx, asyncio, base64
from app.core.interfaces import TranscribeRequest, TranscribeResponse
from app.core.utils import env

class AssemblyAIProvider:
    name = "assemblyai"
    def __init__(self):
        self.api_key = env("ASSEMBLYAI_API_KEY")
        self.base_url = "https://api.assemblyai.com/v2"
        self.headers = {"authorization": self.api_key}

    async def transcribe(self, req: TranscribeRequest) -> TranscribeResponse:
        async with httpx.AsyncClient(timeout=300) as client:
            if req.audio_b64:
                upload_response = await client.post(f"{self.base_url}/upload", headers=self.headers, data=base64.b64decode(req.audio_b64))
                upload_response.raise_for_status()
                audio_url = upload_response.json()['upload_url']
            elif req.audio_url:
                audio_url = req.audio_url
            else:
                raise ValueError("Either audio_url or audio_b64 must be provided.")

            transcript_response = await client.post(f"{self.base_url}/transcript", headers=self.headers, json={"audio_url": audio_url})
            transcript_response.raise_for_status()
            transcript_id = transcript_response.json()['id']

            while True:
                poll_response = await client.get(f"{self.base_url}/transcript/{transcript_id}", headers=self.headers)
                poll_response.raise_for_status()
                result = poll_response.json()
                if result['status'] == 'completed':
                    return TranscribeResponse(text=result['text'], provider=self.name, confidence=result.get('confidence'))
                elif result['status'] == 'error':
                    raise Exception(f"AssemblyAI Error: {result['error']}")
                await asyncio.sleep(3)
EOF

from ..providers.openai_provider import OpenAIProvider
from ..providers.anthropic_provider import AnthropicProvider
from ..providers.assemblyai_provider import AssemblyAIProvider
from ..providers.elevenlabs_provider import ElevenLabsProvider
# Import other providers as they are built

class Router:
    def __init__(self):
        self.providers = {
            "openai": OpenAIProvider(),
            "anthropic": AnthropicProvider(),
            "assemblyai": AssemblyAIProvider(),
            "elevenlabs": ElevenLabsProvider(),
        }
        self.default_text = "openai:gpt-4o"
        self.default_image = "openai:dall-e-3" # Placeholder, assuming OpenAI for now
        self.default_transcribe = "assemblyai"
        self.default_synthesize = "elevenlabs"

    def _get_provider(self, model_string: str, capability: str):
        if ":" in model_string:
            provider_name, _ = model_string.split(":", 1)
        else:
            provider_name = model_string
        
        provider = self.providers.get(provider_name)
        if not provider:
            raise ValueError(f"Provider '{provider_name}' not found for capability '{capability}'.")
        return provider

    async def generate_text(self, req):
        model = req.model or self.default_text
        provider = self._get_provider(model, "generate_text")
        return await provider.generate_text(req)

    async def create_image(self, req):
        model = req.model or self.default_image
        provider = self._get_provider(model, "image")
        return await provider.image(req)

    async def transcribe(self, req):
        provider = self._get_provider(self.default_transcribe, "transcribe")
        return await provider.transcribe(req)

    async def synthesize(self, req):
        provider = self._get_provider(self.default_synthesize, "synthesize")
        return await provider.synthesize(req)

router = Router()
