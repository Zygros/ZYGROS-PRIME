Copy-paste drop-in. This is a universal API plugin pack for OmniWeaver (works with any AI): one interface, one router, many providers. Paste these files into a folder (e.g., omni_plugins/), set your keys in .env, and you can call OpenAI, Anthropic, Vertex, HuggingFace, Replicate, AssemblyAI, ElevenLabs, Stability, Tavus, Azure through a single consistent API.

import os, base64, json, httpx from tenacity import retry, wait_exponential, stop_after_attempt def env(name: str, default: str = "") -> str: val = os.getenv(name, default) if not val: raise RuntimeError(f"Missing env: {name}") return val @retry(wait=wait_exponential(min=0.5, max=8), stop=stop_after_attempt(3)) async def post_json(url: str, headers: dict, json_body: dict, timeout=60): async with httpx.AsyncClient(timeout=timeout) as client: r = await client.post(url, headers=headers, json=json_body) r.raise_for_status() return r.json() @retry(wait=wait_exponential(min=0.5, max=8), stop=stop_after_attempt(3)) async def get_json(url: str, headers: dict, timeout=60): async with httpx.AsyncClient(timeout=timeout) as client: r = await client.get(url, headers=headers) r.raise_for_status() return r.json() def b64(data: bytes) -> str: return base64.b64encode(data).decode("utf-8") 
5) core/router.py
from __future__ import annotations from typing import Dict, Any from pydantic import BaseModel from .interfaces import * from .utils import env from ..providers.openai_llm import OpenAIPlugin from ..providers.anthropic_llm import AnthropicPlugin from ..providers.vertex_ai import VertexPlugin from ..providers.hf_inference import HFPlugin from ..providers.replicate_gen import ReplicatePlugin from ..providers.assemblyai_asr import AssemblyAIPlugin from ..providers.elevenlabs_tts import ElevenLabsPlugin from ..providers.stability_image import StabilityPlugin from ..providers.tavus_video import TavusPlugin from ..providers.azure_cognitive import AzurePlugin class Router: def __init__(self): # register providers self.providers: Dict[str, Plugin] = { "openai": OpenAIPlugin(), "anthropic": AnthropicPlugin(), "vertex": VertexPlugin(), "hf": HFPlugin(), "replicate": ReplicatePlugin(), "assemblyai": AssemblyAIPlugin(), "elevenlabs": ElevenLabsPlugin(), "stability": StabilityPlugin(), "tavus": TavusPlugin(), "azure": AzurePlugin(), } self.default = env("OMNI_DEFAULT_MODEL", "openai:gpt-4o-mini") def _pick(self, target: str | None, capability: Capability) -> Plugin: # target like "openai:gpt-4o-mini" => "openai" if not target: target = self.default provider_name = target.split(":")[0] if provider_name not in self.providers: raise ValueError(f"Unknown provider '{provider_name}'") return self.providers[provider_name] # Unified calls def generate_text(self, req: TextGenRequest) -> TextGenResponse: plugin = self._pick(req.model, "generate_text") return plugin.generate_text(req) def embed(self, req: EmbedRequest) -> EmbedResponse: plugin = self._pick(req.model, "embed") return plugin.embed(req) def transcribe(self, req: TranscribeRequest) -> TranscribeResponse: plugin = self._pick(None, "transcribe") return plugin.transcribe(req) def synthesize(self, req: SynthesizeRequest) -> SynthesizeResponse: plugin = self._pick(req.model, "synthesize") return plugin.synthesize(req) def image(self, req: ImageRequest) -> ImageResponse: plugin = self._pick(req.model, "image") return plugin.image(req) def video(self, req: VideoRequest) -> VideoResponse: plugin = self._pick(req.model, "video") return plugin.video(req) def moderate(self, req: ModerationRequest) -> ModerationResponse: plugin = self._pick(req.__dict__.get("model"), "moderate") return plugin.moderate(req)

providers/openai_llm.py
from __future__ import annotations import os, httpx, json from typing import Dict, Any, List from ..core.interfaces import * from ..core.utils import env class OpenAIPlugin(Plugin): name = "openai" def __init__(self): self.key = env("OPENAI_API_KEY") self.base = os.getenv("OPENAI_BASE_URL","https://api.openai.com/v1") def generate_text(self, req: TextGenRequest) -> TextGenResponse: # Synchronous for simplicity; can convert to async if you prefer headers = {"Authorization": f"Bearer {self.key}", "Content-Type":"application/json"} payload = { "model": (req.model.split(":")[1] if req.model and ":" in req.model else (req.model or "gpt-4o-mini")), "messages": ([{"role":"system","content":req.system}] if req.system else []) + [{"role":"user","content":req.prompt}], "temperature": req.temperature, "max_tokens": req.max_tokens, "response_format": {"type":"json_object"} if req.json_mode else None } # strip None payload = {k:v for k,v in payload.items() if v is not None} r = httpx.post(f"{self.base}/chat/completions", headers=headers, json=payload, timeout=60) r.raise_for_status() j = r.json() text = j["choices"][0]["message"]["content"] return TextGenResponse(text=text, model=j.get("model","openai"), usage=j.get("usage"), provider=self.name) def embed(self, req: EmbedRequest) -> EmbedResponse: headers = {"Authorization": f"Bearer {self.key}", "Content-Type":"application/json"} model = req.model or "text-embedding-3-large" r = httpx.post(f"{self.base}/embeddings", headers=headers, json={"model": model, "input": req.input}, timeout=60) r.raise_for_status() j = r.json() vecs = [d["embedding"] for d in j["data"]] return EmbedResponse(vectors=vecs, model=model, provider=self.name) def moderate(self, req: ModerationRequest) -> ModerationResponse: headers = {"Authorization": f"Bearer {self.key}", "Content-Type":"application/json"} r = httpx.post(f"{self.base}/moderations", headers=headers, json={"model":"omni-moderation-latest","input":req.input}, timeout=60) r.raise_for_status() j = r.json() res = j["results"][0] return ModerationResponse(flagged=res["flagged"], categories=res["categories"], provider=self.name) 
providers/anthropic_llm.py
from __future__ import annotations import httpx, os from ..core.interfaces import * from ..core.utils import env class AnthropicPlugin(Plugin): name = "anthropic" def __init__(self): self.key = env("ANTHROPIC_API_KEY") self.base = os.getenv("ANTHROPIC_BASE","https://api.anthropic.com/v1") def generate_text(self, req: TextGenRequest) -> TextGenResponse: headers = {"x-api-key": self.key, "anthropic-version":"2023-06-01", "content-type":"application/json"} model = (req.model.split(":")[1] if req.model and ":" in req.model else (req.model or "claude-3-5-sonnet-20240620")) messages = [{"role":"user","content":req.prompt}] if req.system: # Claude supports system prompt separately payload = {"model":model,"max_tokens":req.max_tokens,"temperature":req.temperature,"messages":messages,"system":req.system} else: payload = {"model":model,"max_tokens":req.max_tokens,"temperature":req.temperature,"messages":messages} r = httpx.post(f"{self.base}/messages", headers=headers, json=payload, timeout=60) r.raise_for_status() j = r.json() text = "".join([b.get("text","") for b in j["content"] if b.get("type")=="text"]) return TextGenResponse(text=text, model=model, usage=j.get("usage"), provider=self.name) 
providers/vertex_ai.py (server-side SA auth required)
from __future__ import annotations import os, json, httpx, google.auth from google.auth.transport.requests import Request as GoogleRequest from ..core.interfaces import * from ..core.utils import env # Uses REST prediction endpoint for Gemini/GCP (assuming proper IAM) class VertexPlugin(Plugin): name = "vertex" def __init__(self): self.project = env("GOOGLE_VERTEX_PROJECT") self.location = env("GOOGLE_VERTEX_LOCATION") def _id_token(self, audience: str) -> str: creds, _ = google.auth.default() creds.refresh(GoogleRequest()) target = google.auth.transport.requests.Request() return creds.token # for simplicity; in prod generate ID token for audience def generate_text(self, req: TextGenRequest) -> TextGenResponse: model = (req.model.split(":")[1] if req.model and ":" in req.model else (req.model or "gemini-1.5-flash")) endpoint = f"https://{self.location}-aiplatform.googleapis.com/v1/projects/{self.project}/locations/{self.location}/publishers/google/models/{model}:generateContent" token = self._id_token(endpoint) headers = {"Authorization": f"Bearer {token}", "Content-Type":"application/json"} payload = {"contents":[{"role":"user","parts":[{"text":req.prompt}]}], "generationConfig":{"temperature":req.temperature,"maxOutputTokens":req.max_tokens}} r = httpx.post(endpoint, headers=headers, json=payload, timeout=60) r.raise_for_status() j = r.json() text = "".join([p.get("text","") for c in j.get("candidates",[]) for p in c.get("content",{}).get("parts",[])]) return TextGenResponse(text=text, model=model, usage=None, provider=self.name)

8) README.md (how to use)
• pip install -r requirements.txt
• Copy .env.example → .env, add keys.
• Run python demo_all.py to quick-smoke.
• In your app, import Router and call router.generate_text(...), router.transcribe(...), etc.
• Contract: model strings are provider:deployment (e.g., openai:gpt-4o-mini, anthropic:claude-3-5-sonnet, hf:sentence-transformers/...). If no provider: prefix, the router uses defaults.
9) CI AST mapping (plug straight into your command grammar)
Example spoken: “Summarize this and read it in my voice.”
CI AST → calls:
{ "v":"0.9", "mode":"execute", "intent":"run", "object":{"type":"plugin","name":"llm+tts"}, "params":{ "llm_model":"openai:gpt-4o-mini", "tts_model":"elevenlabs:eleven_multilingual_v2", "voice_id":"Rachel", "text":"<user text here>" } } 
Dispatch:
• generate_text(TextGenRequest(...)) → summary
• synthesize(SynthesizeRequest(text=summary, voice_id=...)) → audio_b64
10) Notes
• Some providers (Vertex, Replicate, Tavus) use async jobs / polling; the stubs show the call—wire a poller or webhook for completion.
• Keep secrets in .env or your vault.
• This pack is intentionally minimal + consistent; extend each provider with more endpoints as needed (tools, RAG, batch).
This is the universal plugin pack: single interface, many providers, copy-pasteable, works with any AI. Plug it into OmniWeaver’s router and you’re live across the top APIs.
