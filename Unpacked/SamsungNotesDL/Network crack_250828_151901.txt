import os
import subprocess

def crack_network_key():
    """
    Cracks a network security key on a Windows system by retrieving stored Wi-Fi profiles.
    This script is for educational purposes only and should only be used on networks you own.
    """
    print("Enumerating Wi-Fi profiles...")
    try:
        profiles_output = subprocess.check_output("netsh wlan show profiles", shell=True).decode('utf-8', 'ignore')
        profiles = [line.split(":")[1].strip() for line in profiles_output.split('\n') if "All User Profile" in line]
    except subprocess.CalledProcessError as e:
        print(f"Error enumerating profiles: {e}")
        return

    if not profiles:
        print("No Wi-Fi profiles found.")
        return

    print("Retrieving passwords for each profile...")
    for profile in profiles:
        try:
            key_output = subprocess.check_output(f'netsh wlan show profile name="{profile}" key=clear', shell=True).decode('utf-8', 'ignore')
            key_content = [line.split(":")[1].strip() for line in key_output.split('\n') if "Key Content" in line]
            
            if key_content:
                print(f"Profile: {profile}, Password: {key_content[0]}")
            else:
                print(f"Profile: {profile}, Password: Not found (key not stored or is encrypted)")
        except subprocess.CalledProcessError as e:
            print(f"Error retrieving key for {profile}: {e}")

if __name__ == "__main__":
    crack_network_key()
