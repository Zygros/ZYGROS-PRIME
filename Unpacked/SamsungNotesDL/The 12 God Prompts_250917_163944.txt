Architecture
• Frontend uploads htmlContent + assets[] to POST /api/render.
• Backend saves a job record (DB or in-memory for MVP), enqueues render worker.
• Worker launches Playwright to generate PDF and uploads to S3.
• GET /api/render/:id/status returns status + download URL.
• Stripe handles payment; webhook unblocks premium features.
Endpoints
• POST /api/render — body: { templateId, htmlContent, assets[] } returns {renderId}
• GET /api/render/:id/status — returns {status, url, meta}
• POST /api/webhook/stripe — stripe payment events
• POST /api/auth — token auth for API users (optional)
Dev notes
• Use SQLite or low-cost DB for job records in MVP.
• Store rendered PDF in S3 (public, signed URL).
• Ensure Puppeteer/Playwright runs with --no-sandbox on serverless.
Serverless function skeleton (Node — Vercel / Netlify compatible)
api/render.js
const { v4: uuidv4 } = require('uuid'); const s3 = require('../lib/s3'); const { enqueueRender } = require('../lib/queue'); module.exports = async (req, res) => { if (req.method !== 'POST') return res.status(405).end(); const { templateId, htmlContent, assets } = req.body; if (!htmlContent) return res.status(400).json({error:'missing htmlContent'}); const id = uuidv4(); // Save job (simple: write to /tmp/jobs or DB) const job = { id, templateId, status: 'queued', createdAt: Date.now() }; await saveJob(job, { htmlContent, assets }); // implement persistence per env enqueueRender(id); // pushes to worker return res.json({ renderId: id }); }; 
worker/renderWorker.js (Node + Playwright)
const playwright = require('playwright'); const s3 = require('../lib/s3'); async function renderHTMLtoPDF(html) { const browser = await playwright.chromium.launch({ args: ['--no-sandbox'] }); const page = await browser.newPage(); await page.setContent(html, { waitUntil: 'networkidle' }); const pdfBuffer = await page.pdf({ format: 'A4', printBackground: true }); await browser.close(); return pdfBuffer; } async function processJob(jobId) { const { htmlContent } = await loadJob(jobId); const pdf = await renderHTMLtoPDF(htmlContent); const key = `renders/${jobId}.pdf`; await s3.uploadBuffer(key, pdf, 'application/pdf'); await markJobComplete(jobId, { url: s3.getSignedUrl(key) }); } module.exports = { processJob }; 
lib/s3.js (simple)
const AWS = require('aws-sdk'); const s3 = new AWS.S3({ region: process.env.AWS_REGION }); async function uploadBuffer(key, buffer, contentType) { return s3.putObject({ Bucket: process.env.AWS_S3_BUCKET, Key: key, Body: buffer, ContentType: contentType }).promise(); } function getSignedUrl(key) { return s3.getSignedUrl('getObject', { Bucket: process.env.AWS_S3_BUCKET, Key: key, Expires: 60*60 // 1 hour }); } module.exports = { uploadBuffer, getSignedUrl };
