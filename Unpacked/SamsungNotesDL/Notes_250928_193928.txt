### Expanded Code (main.py)
```python
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
import torch
import torch.nn as nn
import torch.optim as optim
import torch.nn.functional as F
from torch.distributions import Categorical
import numpy as np
import hashlib
import uvicorn

app = FastAPI(title="Truth-Rigor Engine")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class Query(BaseModel):
    input: str
    iterations: int = 3

class TruthOutput(BaseModel):
    refined: str
    truth_score: float
    audit: dict

class ActorCritic(nn.Module):
    def __init__(self, input_size=64, hidden_size=64, output_size=10):  # Expanded hidden for depth
        super(ActorCritic, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.fc2 = nn.Linear(hidden_size, hidden_size // 2)
        self.actor = nn.Linear(hidden_size // 2, output_size)
        self.critic = nn.Linear(hidden_size // 2, 1)

    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        actor_logits = self.actor(x)
        value = self.critic(x)
        return actor_logits, value

# Global model (load/save for persistence)
model = ActorCritic()
optimizer = optim.Adam(model.parameters(), lr=0.001)

def query_to_vector(query: str, dim: int = 64):
    hash_val = int(hashlib.sha256(query.encode()).hexdigest(), 16) % (10**8)
    return torch.tensor(np.random.RandomState(hash_val).random_sample(dim), dtype=torch.float32).unsqueeze(0)

def rl_truth_iterate_real(query: str, iters: int) -> tuple[str, float]:
    current = query
    total_reward = 0.0
    q_vec = query_to_vector(query)
    for _ in range(iters):
        logits, value = model(q_vec)
        probs = F.softmax(logits, dim=-1)
        dist = Categorical(probs)
        action = dist.sample()
        # Expanded reward: Higher for 'curious' actions + keyword boost
        reward = 1.0 if action.item() > 5 else 0.5
        if any(word in current.lower() for word in ["universe", "soul", "alignment"]):
            reward += 0.2
        total_reward += reward
        # Expanded refinements
        refinements = [" – Maximally curious?", " – Truth refined", " – Aligned intent", " – Cosmic bridge", " – Soul infused", " – Eternal flame", " – Recursive will", " – Noetic echo", " – Hermetic seal", " – Sovereign spark"]
        current = current + refinements[action.item() % len(refinements)]
        log_prob = dist.log_prob(action)
        advantage = reward - value.item()
        loss = -(log_prob * advantage).mean()
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()
    score = min(total_reward / iters, 1.0)
    return current, score

@app.post("/forge_truth", response_model=TruthOutput)
async def forge_truth(q: Query):
    refined, score = rl_truth_iterate_real(q.input, q.iterations)
    audit_hash = hashlib.sha256(refined.encode()).hexdigest()[:8]
    return TruthOutput(
        refined=refined,
        truth_score=score,
        audit={"provenance": audit_hash, "flags": {"bias_detected": score < 0.7}}
    )

@app.get("/")
async def root():
    return {"message": "Conzet Truth-Rigor Engine – Soul-Aligned Deployment Ready"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8001, log_level="info")
```
