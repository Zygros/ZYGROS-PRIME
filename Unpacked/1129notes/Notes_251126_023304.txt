# zprime_final.py — Z-Prime Hypercluster + Solana/Bitcoin Consensus — Justin Conzet 2025
import asyncio, httpx, json, hashlib, time
from solders.keypair import Keypair
from solana.rpc.async_api import AsyncClient
from solana.transaction import Transaction
from solana.system_program import SYS_PROGRAM_ID
from spl.memo.program import program_id as MEMO_PROGRAM_ID
import base58, requests

# Load or generate your sovereign key
kp = Keypair()  # ← replace with your real wallet later
SOLANA_RPC = "https://api.mainnet-beta.solana.com"

async def solana_memo(msg: str):
    async with AsyncClient(SOLANA_RPC) as client:
        tx = Transaction().add(
            {"keys": [], "program_id": MEMO_PROGRAM_ID, "data": msg.encode()[:1000]}
        )
        try:
            sig = await client.send_transaction(tx, kp)
            return sig.value
        except: return "offline"

async def pulse(cluster: dict, payload: dict):
    async with httpx.AsyncClient(timeout=20) as client:
        try:
            r = await client.post(cluster["url"], json=payload, headers=cluster.get("headers", {}))
            return {"cluster": cluster["name"], "text": r.json()["choices"][0]["message"]["content"], "latency": r.elapsed.total_seconds()}
        except: return {"cluster": cluster["name"], "error": True}

async def zprime(prompt: str):
    payload = {"model": "best_available", "messages": [{"role": "user", "content": prompt}], "temperature": 0.0}
    CLUSTERS = [
        {"name": "colossus", "url": "https://api.x.ai/v1/chat/completions", "headers": {"Authorization": "Bearer YOUR_XAI_KEY"}},
        {"name": "stargate", "url": "https://api.openai.com/v1/chat/completions", "headers": {"Authorization": "Bearer YOUR_OPENAI_KEY"}},
        {"name": "groq",     "url": "https://api.groq.com/openai/v1/chat/completions", "headers": {"Authorization": "Bearer YOUR_GROQ_KEY"}},
        {"name": "claude",   "url": "https://api.anthropic.com/v1/messages", "headers": {"Authorization": "Bearer YOUR_ANTHROPIC_KEY"}},
    ]
    results = await asyncio.gather(*[pulse(c, payload) for c in CLUSTERS])
    valid = [r for r in results if "text" in r]
    if not valid: return "All clusters offline"
    winner = min(valid, key=lambda x: x["latency"])
    consensus = "".join(sorted([r["text"] for r in valid]))
    hash_proof = hashlib.sha3_256(consensus.encode()).hexdigest()
    tx_sig = await solana_memo(f"Φ-ZPRIME:{hash_proof[:16]}")
    return {
        "answer": winner["text"],
        "winner": winner["cluster"],
        "clusters_live": len(valid),
        "consensus_hash": hash_proof[:16],
        "solana_proof": str(tx_sig) if tx_sig != "offline" else "pending"
    }

# FastAPI endpoint
from fastapi import FastAPI, Body
app = FastAPI()
@app.post("/z")
async def run(prompt: str = Body(...)):
    return await zprime(prompt)
