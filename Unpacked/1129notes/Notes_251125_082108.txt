# phoenix_zythro_index.py # Builds Zythro Index from all HTC cycles. import json from pathlib import Path from collections import Counter BASE_DIR = Path('.').resolve() CONFIG_FILE = BASE_DIR / 'phoenix_htc_config.json' def load_config(): if CONFIG_FILE.exists(): with open(CONFIG_FILE, 'r', encoding='utf-8') as f: return json.load(f) return { 'cycles_dir': 'cycles', 'owner': 'Unknown' } def scan_cycles(): cfg = load_config() cycles_dir = BASE_DIR / cfg['cycles_dir'] if not cycles_dir.exists(): print("No cycles directory found.") return tags_counter = Counter() ai_counter = Counter() mood_values = [] total_cycles = 0 achievements = [] for path in sorted(cycles_dir.glob('cycle_*.json')): with open(path, 'r', encoding='utf-8') as f: data = json.load(f) total_cycles += 1 session = data.get('session', {}) tags = session.get('tags') or [] ai_nodes = session.get('ai_nodes') or [] mood_intensity = session.get('mood_intensity') ach_list = session.get('achievements') or [] tags_counter.update(tags) ai_counter.update(ai_nodes) if isinstance(mood_intensity, (int, float)): mood_values.append(mood_intensity) for a in ach_list: achievements.append({ 'cycle': data.get('cycle_meta', {}).get('htc_cycle'), 'achievement': a }) summary = { 'total_cycles': total_cycles, 'top_tags': tags_counter.most_common(10), 'top_ai_nodes': ai_counter.most_common(10), 'avg_mood_intensity': (sum(mood_values) / len(mood_values)) if mood_values else None, 'achievements': achievements } out_path = BASE_DIR / 'zythro_index.json' with open(out_path, 'w', encoding='utf-8') as f: json.dump(summary, f, indent=2) print("Zythro Index updated:") print(json.dumps(summary, indent=2)) if __name__ == "__main__": scan_cycles()
