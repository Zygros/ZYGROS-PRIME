    def __init__(self):

        self.ledger: List[str] = []
        
    def get_last_seal(self) -> str | None:
        """Retrieves the hash of the last entry, or None if the ledger is empty (H_0)."""
        if not self.ledger:

    def append_ladder_step(self, data_to_seal: Dict, staff: "ThothianCaduceus") -> None:
        """
        1. Serializes the input data.
        2. Appends the previous seal (H_n-1) to the current data payload.
        3. Seals the combined payload to create the new seal (H_n).
        4. Stores H_n in the ledger.

        # Get the hash of the previous step (H_{n-1})
        previous_seal = self.get_last_seal()

            data_to_seal,

        # 2. Append the previous seal to the current payload for chaining
        if previous_seal:
            # For H_n (where n > 0): Payload = data + previous_seal

            # For H_0: Payload = data (The Genesis Block)
            chained_payload = f"{current_data_str}::GENESIS_START"
